\begin{absolutelynopagebreak}

{\PaliGlossA{kathañca, visākhe, gopālakuposatho hoti?}}\\

\setstretch{.5}

\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}

\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}

\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}

\setstretch{.5}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evamevaṃ kho, visākhe, idhekacco uposathiko iti paṭisañcikkhati:}}\\

\setstretch{.5}

\end{addmargin}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘ahaṃ khvajja idañcidañca khādanīyaṃ khādiṃ, idañcidañca bhojanīyaṃ bhuñjiṃ;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}

\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Tomorrow I’ll eat this and that, and have a meal of this and that.’}}\\

\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
