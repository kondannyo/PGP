\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{64. aveccappasannasutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{64. Experiential Confidence}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“ye keci, bhikkhave, mayi aveccappasannā, sabbe te sotāpannā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicants, all those who have experiential confidence in me have entered the stream.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tesaṃ sotāpannānaṃ pañcannaṃ idha niṭṭhā, pañcannaṃ idha vihāya niṭṭhā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Of those who have entered the stream, five conclude their path in this realm, and five conclude their path after leaving this realm behind.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{katamesaṃ pañcannaṃ idha niṭṭhā?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Which five conclude their path in this realm?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sattakkhattuparamassa, kolaṅkolassa, ekabījissa, sakadāgāmissa, yo ca diṭṭheva dhamme arahā—}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{The one who has seven rebirths at most, the one who goes from family to family, the one-seeder, the once returner, and the one who is perfected in the present life.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{imesaṃ pañcannaṃ idha niṭṭhā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{These five conclude their path in this realm.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{katamesaṃ pañcannaṃ idha vihāya niṭṭhā?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Which five conclude their path after leaving this realm behind?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{antarāparinibbāyissa, upahaccaparinibbāyissa, asaṅkhāraparinibbāyissa, sasaṅkhāraparinibbāyissa, uddhaṃsotassa akaniṭṭhagāmino—}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{The one who is extinguished between one life and the next, the one who is extinguished upon landing, the one who is extinguished without extra effort, the one who is extinguished with extra effort, and the one who heads upstream, going to the Akaniṭṭha realm.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{imesaṃ pañcannaṃ idha vihāya niṭṭhā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{These five conclude their path after leaving this realm behind.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ye keci, bhikkhave, mayi aveccappasannā sabbe te sotāpannā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{All those who have experiential confidence in me have entered the stream.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tesaṃ sotāpannānaṃ imesaṃ pañcannaṃ idha niṭṭhā, imesaṃ pañcannaṃ idha vihāya niṭṭhā”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Of those who have entered the stream, these five conclude their path in this realm, and these five conclude their path after leaving this realm behind.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}
