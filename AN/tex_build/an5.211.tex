{\PaliGlossB{211. An Abuser}}\\

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“yo so, bhikkhave, bhikkhu akkosakaparibhāsako ariyūpavādī sabrahmacārīnaṃ, tassa pañca ādīnavā pāṭikaṅkhā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicants, a mendicant who abuses and insults their spiritual companions, speaking ill of the noble ones, can expect these five drawbacks.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{katame pañca?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{What five?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{pārājiko vā hoti chinnaparipantho, aññataraṃ vā saṃkiliṭṭhaṃ āpattiṃ āpajjati, bāḷhaṃ vā rogātaṅkaṃ phusati, sammūḷho kālaṃ karoti, kāyassa bhedā paraṃ maraṇā apāyaṃ duggatiṃ vinipātaṃ nirayaṃ upapajjati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{They’re expelled, cut off, shut out; or they commit a corrupt offense; or they contract a severe illness. They die confused. And when their body breaks up, after death, they’re reborn in a place of loss, a bad place, the underworld, hell.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yo so, bhikkhave, bhikkhu akkosakaparibhāsako ariyūpavādī sabrahmacārīnaṃ, tassa ime pañca ādīnavā pāṭikaṅkhā”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{A mendicant who abuses and insults their spiritual companions, speaking ill of the noble ones, can expect these five drawbacks.”}}\\
\end{addmargin}
