\begin{absolutelynopagebreak}

{\PaliGlossA{“sotānugatānaṃ, bhikkhave, dhammānaṃ, vacasā paricitānaṃ, manasānupekkhitānaṃ, diṭṭhiyā suppaṭividdhānaṃ cattāro ānisaṃsā pāṭikaṅkhā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicants, you can expect four benefits when the teachings have been followed by ear, reinforced by recitation, examined by the mind, and well comprehended theoretically.}}\\
\end{addmargin}

\setstretch{.7}

\begin{absolutelynopagebreak}

\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}

\end{addmargin}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{They’ve followed those teachings by ear, reinforced them by recitation, examined them by the mind, and well comprehended them theoretically.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{so muṭṭhassati kālaṃ kurumāno aññataraṃ devanikāyaṃ upapajjati.}}\\

\begin{absolutelynopagebreak}
