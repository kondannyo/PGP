\begin{absolutelynopagebreak}

{\PaliGlossA{“catasso imā, bhikkhave, dakkhiṇā visuddhiyo.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicants, there are these four ways of purifying a religious donation.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}

\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atthi, bhikkhave, dakkhiṇā dāyakato visujjhati, no paṭiggāhakato;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{There’s a religious donation that’s purified by the giver, not the recipient.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atthi, bhikkhave, dakkhiṇā paṭiggāhakato visujjhati, no dāyakato;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{There’s a religious donation that’s purified by the recipient, not the giver.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atthi, bhikkhave, dakkhiṇā neva dāyakato visujjhati, no paṭiggāhakato;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{There’s a religious donation that’s purified by neither the giver nor the recipient.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{There’s a religious donation that’s purified by both the giver and the recipient.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kathañca, bhikkhave, dakkhiṇā dāyakato visujjhati, no paṭiggāhakato?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{And how is a religious donation purified by the giver, not the recipient?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{idha, bhikkhave, dāyako hoti sīlavā kalyāṇadhammo; paṭiggāhakā honti dussīlā pāpadhammā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It’s when the giver is ethical, of good character, but the recipient is unethical, of bad character.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\setstretch{.5}
{\PaliGlossB{It’s when the giver is unethical, of bad character, but the recipient is ethical, of good character.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}

\end{absolutelynopagebreak}
