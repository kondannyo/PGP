\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}

\setstretch{.5}
{\PaliGlossB{fast, beautiful, and well proportioned?}}\\

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{idha, bhikkhave, bhikkhu āsavānaṃ khayā anāsavaṃ cetovimuttiṃ paññāvimuttiṃ diṭṭheva dhamme sayaṃ abhiññā sacchikatvā upasampajja viharati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It’s a mendicant who realizes the undefiled freedom of heart and freedom by wisdom in this very life. And they live having realized it with their own insight due to the ending of defilements.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\setstretch{.7}
{\PaliGlossA{idamassa javasmiṃ vadāmi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{This is how they’re fast, I say.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{When asked a question about the teaching or training, they answer without faltering.}}\\
\end{addmargin}
\end{absolutelynopagebreak}
