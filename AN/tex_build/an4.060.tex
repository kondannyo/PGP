\begin{addmargin}[1em]{2em}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“catūhi kho, gahapati, dhammehi samannāgato ariyasāvako gihisāmīcipaṭipadaṃ paṭipanno hoti yasopaṭilābhiniṃ saggasaṃvattanikaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Householder, when a noble disciple does four things they are practicing appropriately for a layperson, which brings fame and leads to heaven.}}\\
\end{addmargin}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{What four?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{idha, gahapati, ariyasāvako bhikkhusaṅghaṃ paccupaṭṭhito hoti cīvarena, bhikkhusaṅghaṃ paccupaṭṭhito hoti piṇḍapātena, bhikkhusaṅghaṃ paccupaṭṭhito hoti senāsanena, bhikkhusaṅghaṃ paccupaṭṭhito hoti gilānappaccayabhesajjaparikkhārena.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It’s when a noble disciple serves the mendicant Saṅgha with robes, alms-food, lodgings, and medicines and supplies for the sick.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{imehi kho, gahapati, catūhi dhammehi samannāgato ariyasāvako gihisāmīcipaṭipadaṃ paṭipanno hoti yasopaṭilābhiniṃ saggasaṃvattanikanti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{When a noble disciple does these four things they are practicing appropriately for a layperson, which brings fame and leads to heaven.}}\\
\end{addmargin}
\end{absolutelynopagebreak}
