\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“sattahi, bhikkhave, dhammehi samannāgato bhikkhu nacirasseva āsavānaṃ khayā … pe … sacchikatvā upasampajja vihareyya.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicants, a mendicant with seven qualities soon realizes the supreme culmination of the spiritual path in this very life. They live having achieved with their own insight the goal for which gentlemen rightly go forth from the lay life to homelessness.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}

{\PaliGlossA{idha, bhikkhave, bhikkhu saddho hoti, sīlavā hoti, bahussuto hoti, paṭisallīno hoti, āraddhavīriyo hoti, satimā hoti, paññavā hoti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It’s when a mendicant is faithful, ethical, learned, secluded, energetic, mindful, and wise.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{A mendicant with these seven qualities soon realizes the supreme culmination of the spiritual path in this very life. They live having achieved with their own insight the goal for which gentlemen rightly go forth from the lay life to homelessness.”}}\\
\end{addmargin}
