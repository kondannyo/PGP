\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{    -}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{api ca khvassa tatheva pāpakammaṃ pavedenti:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Rather, the kings are informed of someone’s bad deed:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘ayaṃ puriso surāmerayamajjapamādaṭṭhānaṃ anuyutto itthiṃ vā purisaṃ vā jīvitā voropesi;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘While under the influence of alcoholic drinks that cause negligence, this person murdered a woman or a man.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ayaṃ puriso surāmerayamajjapamādaṭṭhānaṃ anuyutto gāmā vā araññā vā adinnaṃ theyyasaṅkhātaṃ ādiyi;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Or they stole something from a village or wilderness.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ayaṃ puriso surāmerayamajjapamādaṭṭhānaṃ anuyutto paritthīsu parakumārīsu cārittaṃ āpajji;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Or they had sexual relations with women or maidens under someone else’s protection.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ayaṃ puriso surāmerayamajjapamādaṭṭhānaṃ anuyutto gahapatissa vā gahapatiputtassa vā musāvādena atthaṃ pabhañjīti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Or they ruined a householder or householder’s child by lying.’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tamenaṃ rājāno gahetvā surāmerayamajjapamādaṭṭhānahetu hananti vā bandhanti vā pabbājenti vā yathāpaccayaṃ vā karonti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then the kings have them arrested for being under the influence of alcoholic drinks that cause negligence, and execute, imprison, or banish them, or do what the case requires.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{api nu tumhehi evarūpaṃ diṭṭhaṃ vā sutaṃ vā’”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Have you ever seen or heard of such a case?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“diṭṭhañca no, bhante, sutañca suyyissati cā”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Sir, we have seen it and heard of it, and we will hear of it again.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}
