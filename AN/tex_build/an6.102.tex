\setstretch{.5}
{\PaliGlossB{102. Transience}}\\

\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“cha, bhikkhave, ānisaṃse sampassamānena alameva bhikkhunā sabbasaṅkhāresu anodhiṃ karitvā aniccasaññaṃ upaṭṭhāpetuṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicants, seeing six benefits is quite enough to establish the perception of impermanence in all conditions without qualification.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{katame cha?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{What six?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘sabbasaṅkhārā ca me anavatthitā khāyissanti, sabbaloke ca me mano nābhiramissati, sabbalokā ca me mano vuṭṭhahissati, nibbānapoṇañca me mānasaṃ bhavissati, saṃyojanā ca me pahānaṃ gacchissanti, paramena ca sāmaññena samannāgato bhavissāmī’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘All conditions will appear to me as transient.’ ‘My mind will not delight anywhere in the world.’ ‘My mind will rise above the whole world.’ ‘My mind will incline to extinguishment.’ ‘My fetters will be given up.’ ‘I will achieve the ultimate goal of the ascetic life.’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ime kho, bhikkhave, cha ānisaṃse sampassamānena alameva bhikkhunā sabbasaṅkhāresu anodhiṃ karitvā aniccasaññaṃ upaṭṭhāpetun”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Seeing these six benefits is quite enough to establish the perception of impermanence in all conditions without qualification.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}
