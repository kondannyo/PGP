{\PaliGlossA{121. pubbaṅgamasutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{121. Forerunner}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“sūriyassa, bhikkhave, udayato etaṃ pubbaṅgamaṃ etaṃ pubbanimittaṃ, yadidaṃ—aruṇuggaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicants, the dawn is the forerunner and precursor of the sunrise.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evamevaṃ kho, bhikkhave, kusalānaṃ dhammānaṃ etaṃ pubbaṅgamaṃ etaṃ pubbanimittaṃ, yadidaṃ—sammādiṭṭhi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{In the same way right view is the forerunner and precursor of skillful qualities.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sammādiṭṭhikassa, bhikkhave, sammāsaṅkappo pahoti, sammāsaṅkappassa sammāvācā pahoti, sammāvācassa sammākammanto pahoti, sammākammantassa sammāājīvo pahoti, sammāājīvassa sammāvāyāmo pahoti, sammāvāyāmassa sammāsati pahoti, sammāsatissa sammāsamādhi pahoti, sammāsamādhissa sammāñāṇaṃ pahoti, sammāñāṇissa sammāvimutti pahotī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Right view gives rise to right thought. Right thought gives rise to right speech. Right speech gives rise to right action. Right action gives rise to right livelihood. Right livelihood gives rise to right effort. Right effort gives rise to right mindfulness. Right mindfulness gives rise to right immersion. Right immersion gives rise to right knowledge. Right knowledge gives rise to right freedom.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}
