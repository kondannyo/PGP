\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{And I say that these five things are not got by praying or wishing for them.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{imesaṃ kho, gahapati, pañcannaṃ dhammānaṃ iṭṭhānaṃ kantānaṃ manāpānaṃ dullabhānaṃ lokasmiṃ āyācanahetu vā patthanāhetu vā paṭilābho abhavissa, ko idha kena hāyetha?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
