\begin{addmargin}[1em]{2em}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘No matter how this person performs a deed, they experience the result as it should be experienced.’ This being so, the spiritual life can be lived, and there is a chance of making a complete end of suffering.}}\\
\end{addmargin}

\begin{addmargin}[1em]{2em}
