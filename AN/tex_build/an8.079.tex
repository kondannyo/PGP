\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“aṭṭhime, bhikkhave, dhammā sekhassa bhikkhuno parihānāya saṃvattanti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“These eight things lead to the decline of a mendicant trainee.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{katame aṭṭha?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{What eight?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kammārāmatā, bhassārāmatā, niddārāmatā, saṅgaṇikārāmatā, indriyesu aguttadvāratā, bhojane amattaññutā, saṃsaggārāmatā, papañcārāmatā—}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{They relish work, talk, sleep, and company. They don’t guard the sense doors and they eat too much. They relish closeness and proliferation.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}

\setstretch{.5}
{\PaliGlossB{These eight things lead to the decline of a mendicant trainee.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{aṭṭhime, bhikkhave, dhammā sekhassa bhikkhuno aparihānāya saṃvattanti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{These eight things don’t lead to the decline of a mendicant trainee.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\setstretch{.7}

\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{na kammārāmatā, na bhassārāmatā, na niddārāmatā, na saṅgaṇikārāmatā, indriyesu guttadvāratā, bhojane mattaññutā, asaṃsaggārāmatā, nippapañcārāmatā—}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{They don’t relish work, talk, and sleep. They guard the sense doors, and they don’t eat too much. They don’t relish closeness and proliferation.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{These eight things don’t lead to the decline of a mendicant trainee.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}
