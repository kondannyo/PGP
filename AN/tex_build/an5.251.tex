\begin{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}

\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}

\setstretch{.5}
{\PaliGlossB{It’s a mendicant who has the entire spectrum of an adept’s ethics,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{asekhena samādhikkhandhena samannāgato hoti;}}\\

\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}

{\PaliGlossA{asekhena paññākkhandhena samannāgato hoti;}}\\

\setstretch{.5}
{\PaliGlossB{wisdom,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{asekhena vimuttikkhandhena samannāgato hoti;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{freedom,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{and the knowledge and vision of freedom.}}\\
