\begin{addmargin}[1em]{2em}

\begin{addmargin}[1em]{2em}
\setstretch{.5}

\begin{addmargin}[1em]{2em}
\setstretch{.5}

\begin{addmargin}[1em]{2em}

\begin{addmargin}[1em]{2em}
