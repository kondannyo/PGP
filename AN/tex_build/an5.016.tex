\begin{absolutelynopagebreak}

{\PaliGlossA{imesaṃ kho, bhikkhave, pañcannaṃ balānaṃ etaṃ aggaṃ etaṃ saṅgāhikaṃ etaṃ saṅghātaniyaṃ, yadidaṃ paññābalaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Of these five powers, the power of wisdom is the chief. It holds and binds everything together.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{seyyathāpi, bhikkhave, kūṭāgārassa etaṃ aggaṃ etaṃ saṅgāhikaṃ etaṃ saṅghātaniyaṃ, yadidaṃ kūṭaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It’s like a bungalow. The roof-peak is the chief point, which holds and binds everything together.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

{\PaliGlossA{evamevaṃ kho, bhikkhave, imesaṃ pañcannaṃ balānaṃ etaṃ aggaṃ etaṃ saṅgāhikaṃ etaṃ saṅghātaniyaṃ, yadidaṃ paññābalan”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{In the same way, of these five powers, the power of wisdom is the chief. It holds and binds everything together.”}}\\
\end{addmargin}
