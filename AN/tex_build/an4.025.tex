\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“nayidaṃ, bhikkhave, brahmacariyaṃ vussati janakuhanatthaṃ, na janalapanatthaṃ, na lābhasakkārasilokānisaṃsatthaṃ, na itivādappamokkhānisaṃsatthaṃ, na ‘iti maṃ jano jānātū’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicants, this spiritual life is not lived for the sake of deceiving people or flattering them, nor for the benefit of possessions, honor, or popularity, nor for the benefit of winning debates, nor thinking, ‘So let people know about me!’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{This spiritual life is lived for the sake of restraint, giving up, fading away, and cessation.}}\\
\end{addmargin}
\end{absolutelynopagebreak}
