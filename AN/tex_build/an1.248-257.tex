\setstretch{.7}
{\PaliGlossA{“etadaggaṃ, bhikkhave, mama sāvakānaṃ upāsakānaṃ paṭhamaṃ saraṇaṃ gacchantānaṃ yadidaṃ tapussabhallikā vāṇijā. (1)}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“The foremost of my laymen in first going for refuge are the merchants Tapussa and Bhallika.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{… dāyakānaṃ yadidaṃ sudatto gahapati anāthapiṇḍiko. (2)}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{… as a donor is the householder Sudatta Anāthapiṇḍika.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{… dhammakathikānaṃ yadidaṃ citto gahapati macchikāsaṇḍiko. (3)}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{… who speak on the teaching is the householder Citta Macchikāsaṇḍika.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{… catūhi saṅgahavatthūhi parisaṃ saṅgaṇhantānaṃ yadidaṃ hatthako āḷavako. (4)}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{… who attract a congregation by the four ways of being inclusive is Hatthaka Āḷavaka.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{… paṇītadāyakānaṃ yadidaṃ mahānāmo sakko. (5)}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{… who donate fine things is Mahānāma Sakka.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{… manāpadāyakānaṃ yadidaṃ uggo gahapati vesāliko. (6)}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{… who donate nice things is the householder Ugga of Vesālī.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{… saṅghupaṭṭhākānaṃ yadidaṃ hatthigāmako uggato gahapati. (7)}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{… who attend on the Saṅgha is the householder Uggata of Hatthi Village.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{… aveccappasannānaṃ yadidaṃ sūrambaṭṭho. (8)}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{… who have experiential confidence is Sūrambaṭṭha.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{… puggalappasannānaṃ yadidaṃ jīvako komārabhacco. (9)}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{… who have confidence in a person is Jīvaka Komārabhacca.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
