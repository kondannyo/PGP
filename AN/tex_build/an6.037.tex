\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{37. chaḷaṅgadānasutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{37. A Gift With Six Factors}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ekaṃ samayaṃ bhagavā sāvatthiyaṃ viharati jetavane anāthapiṇḍikassa ārāme.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{At one time the Buddha was staying near Sāvatthī in Jeta’s Grove, Anāthapiṇḍika’s monastery.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tena kho pana samayena veḷukaṇḍakī nandamātā upāsikā sāriputtamoggallānappamukhe bhikkhusaṃghe chaḷaṅgasamannāgataṃ dakkhiṇaṃ patiṭṭhāpeti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Now at that time Veḷukaṇṭakī, Nanda’s mother, was preparing a religious donation for the mendicant Saṅgha headed by Sāriputta and Moggallāna.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{addasā kho bhagavā dibbena cakkhunā visuddhena atikkantamānusakena veḷukaṇḍakiṃ nandamātaraṃ upāsikaṃ sāriputtamoggallānappamukhe bhikkhusaṃghe chaḷaṅgasamannāgataṃ dakkhiṇaṃ patiṭṭhāpentiṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{The Buddha saw her doing this, with his clairvoyance that is purified and superhuman,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{disvā bhikkhū āmantesi:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{and he addressed the mendicants:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“esā, bhikkhave, veḷukaṇḍakī nandamātā upāsikā sāriputtamoggallānappamukhe bhikkhusaṃghe chaḷaṅgasamannāgataṃ dakkhiṇaṃ patiṭṭhāpeti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“This Veḷukaṇṭakī, Nanda’s mother, is preparing a religious donation for the mendicant Saṅgha headed by Sāriputta and Moggallāna.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kathañca, bhikkhave, chaḷaṅgasamannāgatā dakkhiṇā hoti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{And how does a religious donation have six factors?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{idha, bhikkhave, dāyakassa tīṇaṅgāni honti, paṭiggāhakānaṃ tīṇaṅgāni.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Three factors apply to the donor and three to the recipients.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{katamāni dāyakassa tīṇaṅgāni?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{What three factors apply to the donor?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{idha, bhikkhave, dāyako pubbeva dānā sumano hoti, dadaṃ cittaṃ pasādeti, datvā attamano hoti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It’s when a donor is in a good mood before giving, while giving they feel confident, and after giving they’re uplifted.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{imāni dāyakassa tīṇaṅgāni.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{These three factors apply to the donor.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{katamāni paṭiggāhakānaṃ tīṇaṅgāni?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{What three factors apply to the recipients?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{idha, bhikkhave, paṭiggāhakā vītarāgā vā honti rāgavinayāya vā paṭipannā, vītadosā vā honti dosavinayāya vā paṭipannā, vītamohā vā honti mohavinayāya vā paṭipannā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It’s when the recipients are free of greed, hate, and delusion, or practicing to be free of them.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{imāni paṭiggāhakānaṃ tīṇaṅgāni.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{These three factors apply to the recipients.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{iti dāyakassa tīṇaṅgāni, paṭiggāhakānaṃ tīṇaṅgāni.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Thus three factors apply to the donor and three to the recipients.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evaṃ kho, bhikkhave, chaḷaṅgasamannāgatā dakkhiṇā hoti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{That’s how a religious donation has six factors.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evaṃ chaḷaṅgasamannāgatāya, bhikkhave, dakkhiṇāya na sukaraṃ puññassa pamāṇaṃ gahetuṃ:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It’s not easy to grasp the merit of such an offering by saying that}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘ettako puññābhisando kusalābhisando sukhassāhāro sovaggiko sukhavipāko saggasaṃvattaniko iṭṭhāya kantāya manāpāya hitāya sukhāya saṃvattatī’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{this is the extent of their overflowing merit, overflowing goodness that nurtures happiness and is conducive to heaven, ripening in happiness and leading to heaven. And it leads to what is likable, desirable, agreeable, to welfare and happiness.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho asaṅkhyeyyo appameyyo mahāpuññakkhandhotveva saṅkhaṃ gacchati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It’s simply reckoned as an incalculable, immeasurable, great mass of merit.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{seyyathāpi, bhikkhave, mahāsamudde na sukaraṃ udakassa pamāṇaṃ gahetuṃ:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It’s like trying to grasp how much water is in the ocean. It’s not easy to say}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘ettakāni udakāḷhakānīti vā ettakāni udakāḷhakasatānīti vā ettakāni udakāḷhakasahassānīti vā ettakāni udakāḷhakasatasahassānī’ti vā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{how many gallons, how many hundreds, thousands, hundreds of thousands of gallons there are.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho asaṅkhyeyyo appameyyo mahāudakakkhandhotveva saṅkhaṃ gacchati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It’s simply reckoned as an incalculable, immeasurable, great mass of water.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evamevaṃ kho, bhikkhave, evaṃ chaḷaṅgasamannāgatāya dakkhiṇāya na sukaraṃ puññassa pamāṇaṃ gahetuṃ:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{In the same way, it’s not easy to grasp the merit of such an offering …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{    -}}\\
