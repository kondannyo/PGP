\begin{addmargin}[1em]{2em}
\setstretch{.5}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{katamañca, bhikkhave, saṃvarappadhānaṃ?}}\\

\setstretch{.5}
{\PaliGlossB{And what, mendicants, is the effort to restrain?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{idha, bhikkhave, bhikkhu anuppannānaṃ pāpakānaṃ akusalānaṃ dhammānaṃ anuppādāya chandaṃ janeti vāyamati vīriyaṃ ārabhati cittaṃ paggaṇhāti padahati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It’s when you generate enthusiasm, try, make an effort, exert the mind, and strive so that bad, unskillful qualities don’t arise.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{idaṃ vuccati, bhikkhave, saṃvarappadhānaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{This is called the effort to restrain.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\end{addmargin}
\end{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It’s when you generate enthusiasm, try, make an effort, exert the mind, and strive so that bad, unskillful qualities are given up.}}\\

\begin{addmargin}[1em]{2em}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It’s when you generate enthusiasm, try, make an effort, exert the mind, and strive so that skillful qualities that have arisen remain, are not lost, but increase, mature, and are fulfilled by development.}}\\
\end{addmargin}
