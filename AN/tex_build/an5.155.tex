\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{suttaṃ, geyyaṃ, veyyākaraṇaṃ, gāthaṃ, udānaṃ, itivuttakaṃ, jātakaṃ, abbhutadhammaṃ, vedallaṃ.}}\\
