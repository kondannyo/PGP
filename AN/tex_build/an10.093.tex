\setstretch{.7}

\begin{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}

\begin{addmargin}[1em]{2em}

\begin{absolutelynopagebreak}

{\PaliGlossB{Such venerables like the quiet, are educated to be quiet, and praise the quiet.}}\\

\begin{addmargin}[1em]{2em}
\setstretch{.5}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
