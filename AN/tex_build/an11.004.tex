\begin{addmargin}[1em]{2em}
\setstretch{.5}

\begin{addmargin}[1em]{2em}
\setstretch{.5}

\setstretch{.5}
{\PaliGlossB{When there is no joy, one who lacks joy has destroyed a vital condition for rapture.}}\\

\begin{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
