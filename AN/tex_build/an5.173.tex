\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“pañcahi, bhikkhave, dhammehi samannāgato upāsako yathābhataṃ nikkhitto evaṃ niraye.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicants, a lay follower with five qualities is cast down to hell.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{katamehi pañcahi?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{What five?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{pāṇātipātī hoti … pe … surāmerayamajjapamādaṭṭhāyī hoti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{They kill living creatures, steal, commit sexual misconduct, lie, and use alcoholic drinks that cause negligence.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{A lay follower with these five qualities is cast down to hell.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{pañcahi, bhikkhave, dhammehi samannāgato upāsako yathābhataṃ nikkhitto evaṃ sagge.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{A lay follower with five qualities is raised up to heaven.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

{\PaliGlossA{katamehi pañcahi?}}\\
\begin{addmargin}[1em]{2em}

\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}

\end{addmargin}
\end{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{A lay follower with these five qualities is raised up to heaven.”}}\\
