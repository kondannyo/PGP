\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{72. ājīvakasutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{72. A Disciple of the Ājīvakas}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ekaṃ samayaṃ āyasmā ānando kosambiyaṃ viharati ghositārāme.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{At one time Venerable Ānanda was staying near Kosambi, in Ghosita’s Monastery.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho aññataro ājīvakasāvako gahapati yenāyasmā ānando tenupasaṅkami; upasaṅkamitvā āyasmantaṃ ānandaṃ abhivādetvā ekamantaṃ nisīdi. ekamantaṃ nisinno kho so ājīvakasāvako gahapati āyasmantaṃ ānandaṃ etadavoca:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then a householder who was a disciple of the <i>Ājīvaka</i> ascetics went up to Venerable Ānanda, bowed, sat down to one side, and said to Ānanda:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“kesaṃ no, bhante ānanda, dhammo svākkhāto?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Sir, whose teaching is well explained?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ke loke suppaṭipannā?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Who in the world is practicing well?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ke loke sukatā”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Who in the world has done well?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“tena hi, gahapati, taññevettha paṭipucchissāmi, yathā te khameyya tathā naṃ byākareyyāsi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Well then, householder, I’ll ask you about this in return, and you can answer as you like.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{taṃ kiṃ maññasi, gahapati,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{What do you think, householder?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ye rāgassa pahānāya dhammaṃ desenti, dosassa pahānāya dhammaṃ desenti, mohassa pahānāya dhammaṃ desenti, tesaṃ dhammo svākkhāto no vā?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Is the teaching of those who teach for giving up greed, hate, and delusion well explained or not?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kathaṃ vā te ettha hotī”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Or how do you see this?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“ye, bhante, rāgassa pahānāya dhammaṃ desenti, dosassa pahānāya dhammaṃ desenti, mohassa pahānāya dhammaṃ desenti, tesaṃ dhammo svākkhāto.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“The teaching of those who teach for giving up greed, hate, and delusion is well explained.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evaṃ me ettha hotī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{That’s how I see it.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“taṃ kiṃ maññasi, gahapati,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“What do you think, householder?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ye rāgassa pahānāya paṭipannā, dosassa pahānāya paṭipannā, mohassa pahānāya paṭipannā, te loke suppaṭipannā no vā?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Are those who practice for giving up greed, hate, and delusion practicing well or not?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kathaṃ vā te ettha hotī”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Or how do you see this?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“ye, bhante, rāgassa pahānāya paṭipannā, dosassa pahānāya paṭipannā, mohassa pahānāya paṭipannā, te loke suppaṭipannā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Those who practice for giving up greed, hate, and delusion are practicing well.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evaṃ me ettha hotī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{That’s how I see it.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“taṃ kiṃ maññasi, gahapati,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“What do you think, householder?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yesaṃ rāgo pahīno ucchinnamūlo tālāvatthukato anabhāvaṅkato āyatiṃ anuppādadhammo, yesaṃ doso pahīno ucchinnamūlo tālāvatthukato anabhāvaṅkato āyatiṃ anuppādadhammo, yesaṃ moho pahīno ucchinnamūlo tālāvatthukato anabhāvaṅkato āyatiṃ anuppādadhammo, te loke sukatā no vā?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Have those who’ve given up greed, hate, and delusion—so they’re cut off at the root, made like a palm stump, obliterated, and unable to arise in the future—done well in the world, or not?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kathaṃ vā te ettha hotī”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Or how do you see this?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“yesaṃ, bhante, rāgo pahīno ucchinnamūlo tālāvatthukato anabhāvaṅkato āyatiṃ anuppādadhammo, yesaṃ doso pahīno … pe … yesaṃ moho pahīno ucchinnamūlo tālāvatthukato anabhāvaṅkato āyatiṃ anuppādadhammo, te loke sukatā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Those who’ve given up greed, hate, and delusion have done well in the world.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evaṃ me ettha hotī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{That’s how I see it.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘ye, bhante, rāgassa pahānāya dhammaṃ desenti, dosassa pahānāya dhammaṃ desenti, mohassa pahānāya dhammaṃ desenti, tesaṃ dhammo svākkhāto’ti.}}\\

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘Those who practice for giving up greed, hate, and delusion are practicing well.’}}\\

\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“acchariyaṃ, bhante, abbhutaṃ, bhante.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“It’s incredible, sir, it’s amazing!}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{na ceva nāma sadhammukkaṃsanā bhavissati, na ca paradhammāpasādanā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{There’s no acclaiming your own teaching or disrespecting someone else’s,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{āyataneva dhammadesanā, attho ca vutto, attā ca anupanīto.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{just teaching what’s relevant in that context. The goal is spoken of, but the self is not involved.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tumhe, bhante ānanda, rāgassa pahānāya dhammaṃ desetha, dosassa … pe … mohassa pahanāya dhammaṃ desetha.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{You, sir, teach Dhamma for giving up greed, hate, and delusion.}}\\

\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tumhākaṃ, bhante ānanda, dhammo svākkhāto.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Your teaching is well explained.}}\\

\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
\setstretch{.5}

\begin{absolutelynopagebreak}
