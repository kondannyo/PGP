\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“yassa kassaci, bhikkhave, bhikkhussa vā bhikkhuniyā vā pañca cetokhilā appahīnā pañca cetasovinibandhā asamucchinnā, tassa yā ratti vā divaso vā āgacchati hāniyeva pāṭikaṅkhā kusalesu dhammesu no vuddhi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
