\setstretch{.5}
{\PaliGlossB{113. Bound for Loss}}\\

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“tayome, bhikkhave, āpāyikā nerayikā idamappahāya.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicants, three kinds of people are bound for a place of loss, bound for hell, if they don’t give up this fault.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{katame tayo?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{What three?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yo ca abrahmacārī brahmacāripaṭiñño,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Someone who is unchaste, but claims to be celibate;}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yo ca suddhaṃ brahmacariyaṃ carantaṃ amūlakena abrahmacariyena anuddhaṃseti,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{someone who makes a groundless accusation of unchastity against a person whose celibacy is pure;}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yo cāyaṃ evaṃvādī evaṃdiṭṭhi: ‘natthi kāmesu doso’ti, so tāya kāmesu pātabyataṃ āpajjati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{and someone who has the view, ‘There is nothing wrong with sensual pleasures,’ so they throw themselves into sensual pleasures.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ime kho, bhikkhave, tayo āpāyikā nerayikā idamappahāyā”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{These are the three kinds of people bound for a place of loss, bound for hell, if they don’t give up this fault.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}
