\setstretch{.5}
{\PaliGlossB{Long life, beauty, happiness, and strength.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\setstretch{.5}

\end{addmargin}
\end{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
