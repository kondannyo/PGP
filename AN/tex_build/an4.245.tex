\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{245. sikkhānisaṃsasutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{245. The Benefits of Training}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“sikkhānisaṃsamidaṃ, bhikkhave, brahmacariyaṃ vussati paññuttaraṃ vimuttisāraṃ satādhipateyyaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicants, living this spiritual life is benefited by training and overseen by wisdom; freedom is its heart, and mindfulness is in charge.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kathañca, bhikkhave, sikkhānisaṃsaṃ hoti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{And how is it benefited by training?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{idha, bhikkhave, mayā sāvakānaṃ ābhisamācārikā sikkhā paññattā appasannānaṃ pasādāya pasannānaṃ bhiyyobhāvāya.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Firstly, I laid down for my disciples the training that deals with supplementary regulations in order to inspire confidence in those without it and to increase confidence in those who have it.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yathā yathā, bhikkhave, mayā sāvakānaṃ ābhisamācārikā sikkhā paññattā appasannānaṃ pasādāya pasannānaṃ bhiyyobhāvāya tathā tathā so tassā sikkhāya akhaṇḍakārī hoti acchiddakārī asabalakārī akammāsakārī, samādāya sikkhati sikkhāpadesu.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{They undertake whatever supplementary regulations I have laid down, keeping them unbroken, impeccable, spotless, and unmarred.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{puna caparaṃ, bhikkhave, mayā sāvakānaṃ ādibrahmacariyikā sikkhā paññattā sabbaso sammā dukkhakkhayāya.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Furthermore, I laid down for my disciples the training that deals with the fundamentals of the spiritual life in order to rightly end suffering in every way.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yathā yathā, bhikkhave, mayā sāvakānaṃ ādibrahmacariyikā sikkhā paññattā sabbaso sammā dukkhakkhayāya tathā tathā so tassā sikkhāya akhaṇḍakārī hoti acchiddakārī asabalakārī akammāsakārī, samādāya sikkhati sikkhāpadesu.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{They undertake whatever training that deals with the fundamentals of the spiritual life I have laid down, keeping it unbroken, impeccable, spotless, and unmarred.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evaṃ kho, bhikkhave, sikkhānisaṃsaṃ hoti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{That’s how it’s benefited by training.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kathañca, bhikkhave, paññuttaraṃ hoti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{And how is it overseen by wisdom?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{idha, bhikkhave, mayā sāvakānaṃ dhammā desitā sabbaso sammā dukkhakkhayāya.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{I taught the Dhamma to my disciples in order to rightly end suffering in every way.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yathā yathā, bhikkhave, mayā sāvakānaṃ dhammā desitā sabbaso sammā dukkhakkhayāya tathā tathāssa te dhammā paññāya samavekkhitā honti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{They examine with wisdom any teachings I taught them.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evaṃ kho, bhikkhave, paññuttaraṃ hoti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{That’s how it’s overseen by wisdom.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kathañca, bhikkhave, vimuttisāraṃ hoti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{And how does it have freedom at its heart?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{idha, bhikkhave, mayā sāvakānaṃ dhammā desitā sabbaso sammā dukkhakkhayāya.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{I taught the Dhamma to my disciples in order to rightly end suffering in every way.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yathā yathā, bhikkhave, mayā sāvakānaṃ dhammā desitā sabbaso sammā dukkhakkhayāya tathā tathāssa te dhammā vimuttiyā phusitā honti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{They experience through freedom any teachings I taught them.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evaṃ kho, bhikkhave, vimuttisāraṃ hoti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{That’s how it has freedom at its heart.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kathañca, bhikkhave, satādhipateyyaṃ hoti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{And how is mindfulness in charge?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘iti aparipūraṃ vā ābhisamācārikaṃ sikkhaṃ paripūressāmi, paripūraṃ vā ābhisamācārikaṃ sikkhaṃ tattha tattha paññāya anuggahessāmī’ti—ajjhattaṃyeva sati sūpaṭṭhitā hoti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Mindfulness is well established in oneself: ‘In this way I’ll fulfill the training dealing with supplementary regulations, or support with wisdom in every situation the training dealing with supplementary regulations I’ve already fulfilled.’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘iti aparipūraṃ vā ādibrahmacariyikaṃ sikkhaṃ paripūressāmi, paripūraṃ vā ādibrahmacariyikaṃ sikkhaṃ tattha tattha paññāya anuggahessāmī’ti—ajjhattaṃyeva sati sūpaṭṭhitā hoti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Mindfulness is well established in oneself: ‘In this way I’ll fulfill the training dealing with the fundamentals of the spiritual life, or support with wisdom in every situation the training dealing with the fundamentals of the spiritual life I’ve already fulfilled.’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘iti asamavekkhitaṃ vā dhammaṃ paññāya samavekkhissāmi, samavekkhitaṃ vā dhammaṃ tattha tattha paññāya anuggahessāmī’ti—ajjhattaṃyeva sati sūpaṭṭhitā hoti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Mindfulness is well established in oneself: ‘In this way I’ll examine with wisdom the teaching that I haven’t yet examined, or support with wisdom in every situation the teaching I’ve already examined.’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘iti aphusitaṃ vā dhammaṃ vimuttiyā phusissāmi, phusitaṃ vā dhammaṃ tattha tattha paññāya anuggahessāmī’ti—ajjhattaṃyeva sati sūpaṭṭhitā hoti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Mindfulness is well established in oneself: ‘In this way I’ll experience through freedom the teaching that I haven’t yet experienced, or support with wisdom in every situation the teaching I’ve already experienced.’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evaṃ kho, bhikkhave, satādhipateyyaṃ hoti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{That’s how mindfulness is in charge.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘sikkhānisaṃsamidaṃ, bhikkhave, brahmacariyaṃ vussati paññuttaraṃ vimuttisāraṃ satādhipateyyan’ti,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘Living this spiritual life is benefited by training and overseen by wisdom; freedom is its heart, and mindfulness is in charge.’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{iti yaṃ taṃ vuttaṃ idametaṃ paṭicca vuttan”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{That’s what I said, and this is why I said it.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}
