\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{103. micchattasutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{103. The Wrong Way}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“micchattaṃ, bhikkhave, āgamma virādhanā hoti, no ārādhanā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicants, relying on the wrong way leads to failure, not success.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kathañca, bhikkhave, micchattaṃ āgamma virādhanā hoti, no ārādhanā?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{And how does relying on the wrong way lead to failure, not success?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{micchādiṭṭhikassa, bhikkhave, micchāsaṅkappo pahoti, micchāsaṅkappassa micchāvācā pahoti, micchāvācassa micchākammanto pahoti, micchākammantassa micchāājīvo pahoti, micchāājīvassa micchāvāyāmo pahoti, micchāvāyāmassa micchāsati pahoti, micchāsatissa micchāsamādhi pahoti, micchāsamādhissa micchāñāṇaṃ pahoti, micchāñāṇissa micchāvimutti pahoti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Wrong view gives rise to wrong thought. Wrong thought gives rise to wrong speech. Wrong speech gives rise to wrong action. Wrong action gives rise to wrong livelihood. Wrong livelihood gives rise to wrong effort. Wrong effort gives rise to wrong mindfulness. Wrong mindfulness gives rise to wrong immersion. Wrong immersion gives rise to wrong knowledge. Wrong knowledge gives rise to wrong freedom.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evaṃ kho, bhikkhave, micchattaṃ āgamma virādhanā hoti, no ārādhanā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{That’s how relying on the wrong way leads to failure, not success.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sammattaṃ, bhikkhave, āgamma ārādhanā hoti, no virādhanā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Relying on the right way leads to success, not failure.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kathañca, bhikkhave, sammattaṃ āgamma ārādhanā hoti, no virādhanā?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{And how does relying on the right way lead to success, not failure?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sammādiṭṭhikassa, bhikkhave, sammāsaṅkappo pahoti, sammāsaṅkappassa sammāvācā pahoti, sammāvācassa sammākammanto pahoti, sammākammantassa sammāājīvo pahoti, sammāājīvassa sammāvāyāmo pahoti, sammāvāyāmassa sammāsati pahoti, sammāsatissa sammāsamādhi pahoti, sammāsamādhissa sammāñāṇaṃ pahoti, sammāñāṇissa sammāvimutti pahoti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Right view gives rise to right thought. Right thought gives rise to right speech. Right speech gives rise to right action. Right action gives rise to right livelihood. Right livelihood gives rise to right effort. Right effort gives rise to right mindfulness. Right mindfulness gives rise to right immersion. Right immersion gives rise to right knowledge. Right knowledge gives rise to right freedom.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evaṃ kho, bhikkhave, sammattaṃ āgamma ārādhanā hoti, no virādhanā”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{That’s how relying on the right way leads to success, not failure.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}
