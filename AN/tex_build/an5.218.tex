\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{218. dutiyaapāsādikasutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{218. Uninspiring Conduct (2nd)}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“pañcime, bhikkhave, ādīnavā apāsādike.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicants, there are these five drawbacks of uninspiring conduct.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{katame pañca?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{What five?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{appasannā nappasīdanti, pasannānañca ekaccānaṃ aññathattaṃ hoti, satthusāsanaṃ akataṃ hoti, pacchimā janatā diṭṭhānugatiṃ āpajjati, cittamassa nappasīdati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{You don’t inspire confidence in those without it. You cause some with confidence to change their minds. You don’t follow the Teacher’s instructions. Those who come after you follow your example. And your mind doesn’t become clear.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ime kho, bhikkhave, pañca ādīnavā apāsādike.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{These are the five drawbacks of uninspiring conduct.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{pañcime, bhikkhave, ānisaṃsā pāsādike.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{There are these five benefits of inspiring conduct.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{katame pañca?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{What five?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{appasannā pasīdanti, pasannānañca bhiyyobhāvo hoti, satthusāsanaṃ kataṃ hoti, pacchimā janatā diṭṭhānugatiṃ āpajjati, cittamassa pasīdati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{You inspire confidence in those without it. You increase confidence in those who have it. You follow the Teacher’s instructions. Those who come after you follow your example. And your mind becomes clear.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ime kho, bhikkhave, pañca ānisaṃsā pāsādike”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{These are the five benefits of inspiring conduct.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}
