\setstretch{.5}

\begin{absolutelynopagebreak}
\setstretch{.7}

\setstretch{.5}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{‘kinti me sāvakā dhammadāyādā bhaveyyuṃ, no āmisadāyādā’ti.}}\\

\setstretch{.5}
{\EnglishColumn{‘How can my disciples become heirs in the teaching, not in material things?’}}\\

\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}

\setstretch{.5}
{\EnglishColumn{‘The Teacher’s disciples live as heirs in material things, not in the teaching.’}}\\

\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{ahampi tena ādiyo bhaveyyaṃ:}}\\

\setstretch{.5}
{\EnglishColumn{And they’ll point to me, saying,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{‘āmisadāyādā satthusāvakā viharanti, no dhammadāyādā’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{‘The Teacher’s disciples live as heirs in material things, not in the teaching.’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{Tumhe ca me, bhikkhave, dhammadāyādā bhaveyyātha, no āmisadāyādā, tumhepi tena na ādiyā bhaveyyātha:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{If you become heirs in the teaching, not in material things, they’ll point to you, saying,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{‘dhammadāyādā satthusāvakā viharanti, no āmisadāyādā’ti;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{‘The Teacher’s disciples live as heirs in the teaching, not in material things.’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{ahampi tena na ādiyo bhaveyyaṃ:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{And they’ll point to me, saying,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{‘dhammadāyādā satthusāvakā viharanti, no āmisadāyādā’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{‘The Teacher’s disciples live as heirs in the teaching, not in material things.’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{Tasmātiha me, bhikkhave, dhammadāyādā bhavatha, mā āmisadāyādā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{So, mendicants, be my heirs in the teaching, not in material things.}}\\

{\EnglishColumn{Out of compassion for you, I think,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{‘kinti me sāvakā dhammadāyādā bhaveyyuṃ, no āmisadāyādā’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{‘How can my disciples become heirs in the teaching, not in material things?’}}\\

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{3. Idhāhaṃ, bhikkhave, bhuttāvī assaṃ pavārito paripuṇṇo pariyosito suhito yāvadattho;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{Suppose that I had eaten and refused more food, being full, and having had as much as I needed.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{siyā ca me piṇḍapāto atirekadhammo chaḍḍanīyadhammo.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{And there was some extra alms-food that was going to be thrown away.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{Atha dve bhikkhū āgaccheyyuṃ jighacchādubbalyaparetā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{Then two mendicants were to come who were weak with hunger.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{Tyāhaṃ evaṃ vadeyyaṃ:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{I’d say to them,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{‘ahaṃ khomhi, bhikkhave, bhuttāvī pavārito paripuṇṇo pariyosito suhito yāvadattho;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{‘Mendicants, I have eaten and refused more food, being full, and having had as much as I need.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{atthi ca me ayaṃ piṇḍapāto atirekadhammo chaḍḍanīyadhammo.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{And there is this extra alms-food that’s going to be thrown away.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{Sace ākaṅkhatha, bhuñjatha, no ce tumhe bhuñjissatha, idānāhaṃ appaharite vā chaḍḍessāmi, appāṇake vā udake opilāpessāmī’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{Eat it if you like. Otherwise I’ll throw it out where there is little that grows, or drop it into water that has no living creatures.’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{Tatrekassa bhikkhuno evamassa:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{Then one of those mendicants thought,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{‘bhagavā kho bhuttāvī pavārito paripuṇṇo pariyosito suhito yāvadattho;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{‘The Buddha has eaten and refused more food.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{atthi cāyaṃ bhagavato piṇḍapāto atirekadhammo chaḍḍanīyadhammo.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{And he has some extra alms-food that’s going to be thrown away.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{Sace mayaṃ na bhuñjissāma, idāni bhagavā appaharite vā chaḍḍessati, appāṇake vā udake opilāpessati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{If we don’t eat it he’ll throw it away.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{Vuttaṃ kho panetaṃ bhagavatā:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{But the Buddha has also said:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{“dhammadāyādā me, bhikkhave, bhavatha, mā āmisadāyādā”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{“Be my heirs in the teaching, not in material things.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{Āmisaññataraṃ kho panetaṃ, yadidaṃ piṇḍapāto.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{And alms-food is a kind of material thing.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{Yannūnāhaṃ imaṃ piṇḍapātaṃ abhuñjitvā imināva jighacchādubbalyena evaṃ imaṃ rattindivaṃ vītināmeyyan’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{Instead of eating this alms-food, why don’t I spend this day and night weak with hunger?’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{So taṃ piṇḍapātaṃ abhuñjitvā teneva jighacchādubbalyena evaṃ taṃ rattindivaṃ vītināmeyya.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{And that’s what they did.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{Atha dutiyassa bhikkhuno evamassa:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{Then the second of those mendicants thought,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{‘bhagavā kho bhuttāvī pavārito paripuṇṇo pariyosito suhito yāvadattho;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{‘The Buddha has eaten and refused more food.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{atthi cāyaṃ bhagavato piṇḍapāto atirekadhammo chaḍḍanīyadhammo.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{And he has some extra alms-food that’s going to be thrown away.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{Sace mayaṃ na bhuñjissāma, idāni bhagavā appaharite vā chaḍḍessati, appāṇake vā udake opilāpessati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{If we don’t eat it he’ll throw it away.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{Yannūnāhaṃ imaṃ piṇḍapātaṃ bhuñjitvā jighacchādubbalyaṃ paṭivinodetvā evaṃ imaṃ rattindivaṃ vītināmeyyan’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{Why don’t I eat this alms-food, then spend the day and night having got rid of my hunger and weakness?’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{So taṃ piṇḍapātaṃ bhuñjitvā jighacchādubbalyaṃ paṭivinodetvā evaṃ taṃ rattindivaṃ vītināmeyya.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{And that’s what they did.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{Kiñcāpi so, bhikkhave, bhikkhu taṃ piṇḍapātaṃ bhuñjitvā jighacchādubbalyaṃ paṭivinodetvā evaṃ taṃ rattindivaṃ vītināmeyya, atha kho asuyeva me purimo bhikkhu pujjataro ca pāsaṃsataro ca.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{Even though that mendicant, after eating the alms-food, spent the day and night rid of hunger and weakness, it is the former mendicant who is more worthy of respect and praise.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}

\end{absolutelynopagebreak}
