\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Dīghanakhasutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{With Dīghanakha}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\vskip 0.05in
\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Evaṃ me sutaṃ—}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{So I have heard.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ekaṃ samayaṃ bhagavā rājagahe viharati gijjhakūṭe pabbate sūkarakhatāyaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{At one time the Buddha was staying near Rājagaha, on the Vulture’s Peak Mountain in the Boar’s Cave.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\vskip 0.05in
\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Atha kho dīghanakho paribbājako yena bhagavā tenupasaṅkami; upasaṅkamitvā bhagavatā saddhiṃ sammodi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then the wanderer Dīghanakha went up to the Buddha, and exchanged greetings with him.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Sammodanīyaṃ kathaṃ sāraṇīyaṃ vītisāretvā ekamantaṃ aṭṭhāsi. Ekamantaṃ ṭhito kho dīghanakho paribbājako bhagavantaṃ etadavoca:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{When the greetings and polite conversation were over, he stood to one side, and said to the Buddha,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“ahañhi, bho gotama, evaṃvādī evaṃdiṭṭhi:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Master Gotama, this is my doctrine and view:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘sabbaṃ me nakkhamatī’”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘I believe in nothing.’”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“Yāpi kho te esā, aggivessana, diṭṭhi:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“This view of yours, Aggivessana—}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘sabbaṃ me nakkhamatī’ti, esāpi te diṭṭhi nakkhamatī”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{do you believe in that?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“Esā ce me, bho gotama, diṭṭhi khameyya, taṃpassa tādisameva, taṃpassa tādisamevā”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“If I believed in this view, Master Gotama, it wouldn’t make any difference, it wouldn’t make any difference!”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\vskip 0.05in
\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“Ato kho te, aggivessana, bahū hi bahutarā lokasmiṃ ye evamāhaṃsu:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Well, Aggivessana, there are many more in the world who say,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘taṃpassa tādisameva, taṃpassa tādisamevā’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘It wouldn’t make any difference! It wouldn’t make any difference!’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Te tañceva diṭṭhiṃ nappajahanti aññañca diṭṭhiṃ upādiyanti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{But they don’t give up that view, and they grasp another view.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Ato kho te, aggivessana, tanū hi tanutarā lokasmiṃ ye evamāhaṃsu:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{And there are a scant few in the world who say,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘taṃpassa tādisameva, taṃpassa tādisamevā’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘It wouldn’t make any difference! It wouldn’t make any difference!’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Te tañceva diṭṭhiṃ pajahanti aññañca diṭṭhiṃ na upādiyanti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{And they give up that view by not grasping another view.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}

{\PaliGlossA{Santaggivessana, eke samaṇabrāhmaṇā evaṃvādino evaṃdiṭṭhino:}}\\
\begin{addmargin}[1em]{2em}

\end{absolutelynopagebreak}

\end{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
\setstretch{.5}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tesamayaṃ diṭṭhi sārāgāya santike, saññogāya santike, abhinandanāya santike, ajjhosānāya santike, upādānāya santike;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{is close to greed, bondage, approving, attachment, and grasping.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tatraggivessana ye te samaṇabrāhmaṇā evaṃvādino evaṃdiṭṭhino: ‘sabbaṃ me nakkhamatī’ti}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{The view of the ascetics and brahmins who believe in nothing}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tesamayaṃ diṭṭhi asārāgāya santike, asaññogāya santike, anabhinandanāya santike, anajjhosānāya santike, anupādānāya santike”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{is far from greed, bondage, approving, attachment, and grasping.”}}\\

\setstretch{.5}

\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{believe in some things and not in others.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Yā hi tesaṃ khamati sāyaṃ diṭṭhi sārāgāya santike, saññogāya santike, abhinandanāya santike, ajjhosānāya santike, upādānāya santike;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Their view of what they believe in is close to greed, bondage, approving, attachment, and grasping.}}\\

\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yā hi tesaṃ nakkhamati sāyaṃ diṭṭhi asārāgāya santike, asaññogāya santike, anabhinandanāya santike, anajjhosānāya santike, anupādānāya santike.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Their view of what they don’t believe in is far from greed, bondage, approving, attachment, and grasping.}}\\
\end{addmargin}
\end{absolutelynopagebreak}
