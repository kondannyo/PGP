\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“Ayaṃ kho, āvuso kaccāna, hetu ayaṃ paccayo, yena tāsaṃ devatānaṃ ekaṃ devanikāyaṃ upapannānaṃ santettha ekaccā devatā parittābhā, santi panettha ekaccā devatā appamāṇābhā”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“This is the cause, Reverend Kaccāna, this is the reason why, when those deities have been reborn in a single order of gods, some deities there have limited radiance, while some have limitless radiance.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\vskip 0.05in
\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{15. “Sādhu, bhante anuruddha.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Good, Venerable Anuruddha!}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Atthi ca me ettha uttariṃ paṭipucchitabbaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{I have a further question about this.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Yāvatā, bhante, devatā ābhā sabbā tā saṃkiliṭṭhābhā udāhu santettha ekaccā devatā parisuddhābhā”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Do all the radiant deities have corrupted radiance, or do some there have pure radiance?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“Tadaṅgena kho, āvuso kaccāna, santettha ekaccā devatā saṃkiliṭṭhābhā, santi panettha ekaccā devatā parisuddhābhā”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“In that respect, Reverend Kaccāna, some deities there have corrupted radiance, while some have pure radiance.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\vskip 0.05in
\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{16. “Ko nu kho, bhante, anuruddha, hetu ko paccayo, yena tāsaṃ devatānaṃ ekaṃ devanikāyaṃ upapannānaṃ santettha ekaccā devatā saṃkiliṭṭhābhā, santi panettha ekaccā devatā parisuddhābhā”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“What is the cause, Venerable Anuruddha, what is the reason why, when those deities have been reborn in a single order of gods, some deities there have corrupted radiance, while some have pure radiance?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“Tena hāvuso kaccāna, upamaṃ te karissāmi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Well then, Reverend Kaccāna, I shall give you a simile.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Upamāyapidhekacce viññū purisā bhāsitassa atthaṃ ājānanti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{For by means of a simile some sensible people understand the meaning of what is said.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Seyyathāpi, āvuso kaccāna, telappadīpassa jhāyato telampi aparisuddhaṃ vaṭṭipi aparisuddhā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Suppose an oil lamp was burning with impure oil and impure wick.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{So telassapi aparisuddhattā vaṭṭiyāpi aparisuddhattā andhandhaṃ viya jhāyati;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Because of the impurity of the oil and the wick it burns dimly, as it were.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evameva kho, āvuso kaccāna, idhekacco bhikkhu ‘saṃkiliṭṭhābhā’ti pharitvā adhimuccitvā viharati,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{In the same way, take some mendicant who meditates determined on pervading ‘corrupted radiance’.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tassa kāyaduṭṭhullampi na suppaṭippassaddhaṃ hoti, thinamiddhampi na susamūhataṃ hoti, uddhaccakukkuccampi na suppaṭivinītaṃ hoti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Their physical discomfort is not completely settled, their dullness and drowsiness is not completely eradicated, and their restlessness and remorse is not completely eliminated.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{So kāyaduṭṭhullassapi na suppaṭippassaddhattā thinamiddhassapi na susamūhatattā uddhaccakukkuccassapi na suppaṭivinītattā andhandhaṃ viya jhāyati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Because of this they practice absorption dimly, as it were.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{So kāyassa bhedā paraṃ maraṇā saṃkiliṭṭhābhānaṃ devānaṃ sahabyataṃ upapajjati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{When their body breaks up, after death, they’re reborn in the company of the gods of corrupted radiance.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Seyyathāpi, āvuso kaccāna, telappadīpassa jhāyato telampi parisuddhaṃ vaṭṭipi parisuddhā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Suppose an oil lamp was burning with pure oil and pure wick.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{So telassapi parisuddhattā vaṭṭiyāpi parisuddhattā na andhandhaṃ viya jhāyati;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Because of the purity of the oil and the wick it doesn’t burn dimly, as it were.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evameva kho, āvuso kaccāna, idhekacco bhikkhu ‘parisuddhābhā’ti pharitvā adhimuccitvā viharati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{In the same way, take some mendicant who meditates determined on pervading ‘pure radiance’.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Tassa kāyaduṭṭhullampi suppaṭippassaddhaṃ hoti, thinamiddhampi susamūhataṃ hoti, uddhaccakukkuccampi suppaṭivinītaṃ hoti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Their physical discomfort is completely settled, their dullness and drowsiness is completely eradicated, and their restlessness and remorse is completely eliminated.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{So kāyaduṭṭhullassapi suppaṭippassaddhattā thinamiddhassapi susamūhatattā uddhaccakukkuccassapi suppaṭivinītattā na andhandhaṃ viya jhāyati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Because of this they don’t practice absorption dimly, as it were.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{So kāyassa bhedā paraṃ maraṇā parisuddhābhānaṃ devānaṃ sahabyataṃ upapajjati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{When their body breaks up, after death, they’re reborn in the company of the gods of pure radiance.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Ayaṃ kho, āvuso kaccāna, hetu ayaṃ paccayo yena tāsaṃ devatānaṃ ekaṃ devanikāyaṃ upapannānaṃ santettha ekaccā devatā saṃkiliṭṭhābhā, santi panettha ekaccā devatā parisuddhābhā”ti.}}\\
