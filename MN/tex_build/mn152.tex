\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Indriyabhāvanāsutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{The Development of the Faculties}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\vskip 0.05in
\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Evaṃ me sutaṃ—}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{So I have heard.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ekaṃ samayaṃ bhagavā gajaṅgalāyaṃ viharati suveḷuvane.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{At one time the Buddha was staying near Kajaṅgalā in a bamboo grove.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\vskip 0.05in
\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Atha kho uttaro māṇavo pārāsiviyantevāsī yena bhagavā tenupasaṅkami; upasaṅkamitvā bhagavatā saddhiṃ sammodi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then the brahmin student Uttara, a pupil of the brahmin Pārāsariya, approached the Buddha, and exchanged greetings with him.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Sammodanīyaṃ kathaṃ sāraṇīyaṃ vītisāretvā ekamantaṃ nisīdi. Ekamantaṃ nisinnaṃ kho uttaraṃ māṇavaṃ pārāsiviyantevāsiṃ bhagavā etadavoca:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{When the greetings and polite conversation were over, he sat down to one side. The Buddha said to him,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“deseti, uttara, pārāsiviyo brāhmaṇo sāvakānaṃ indriyabhāvanan”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Uttara, does Pārāsariya teach his disciples the development of the faculties?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“Deseti, bho gotama, pārāsiviyo brāhmaṇo sāvakānaṃ indriyabhāvanan”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“He does, Master Gotama.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“Yathā kathaṃ pana, uttara, deseti pārāsiviyo brāhmaṇo sāvakānaṃ indriyabhāvanan”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“But how does he teach it?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“Idha, bho gotama, cakkhunā rūpaṃ na passati, sotena saddaṃ na suṇāti—}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Master Gotama, it’s when the eye sees no sight and the ear hears no sound.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evaṃ kho, bho gotama, deseti pārāsiviyo brāhmaṇo sāvakānaṃ indriyabhāvanan”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{That’s how Pārāsariya teaches his disciples the development of the faculties.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“Evaṃ sante kho, uttara, andho bhāvitindriyo bhavissati, badhiro bhāvitindriyo bhavissati;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“In that case, Uttara, a blind person and a deaf person will have developed faculties}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yathā pārāsiviyassa brāhmaṇassa vacanaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{according to what Pārāsariya says.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Andho hi, uttara, cakkhunā rūpaṃ na passati, badhiro sotena saddaṃ na suṇātī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{For a blind person sees no sight with the eye and a deaf person hears no sound with the ear.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Evaṃ vutte, uttaro māṇavo pārāsiviyantevāsī tuṇhībhūto maṅkubhūto pattakkhandho adhomukho pajjhāyanto appaṭibhāno nisīdi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{When he said this, Uttara sat silent, embarrassed, shoulders drooping, downcast, depressed, with nothing to say.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\vskip 0.05in
\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Atha kho bhagavā uttaraṃ māṇavaṃ pārāsiviyantevāsiṃ tuṇhībhūtaṃ maṅkubhūtaṃ pattakkhandhaṃ adhomukhaṃ pajjhāyantaṃ appaṭibhānaṃ viditvā āyasmantaṃ ānandaṃ āmantesi:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Knowing this, the Buddha addressed Venerable Ānanda,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“aññathā kho, ānanda, deseti pārāsiviyo brāhmaṇo sāvakānaṃ indriyabhāvanaṃ, aññathā ca panānanda, ariyassa vinaye anuttarā indriyabhāvanā hotī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Ānanda, the development of the faculties taught by Pārāsariya is quite different from the supreme development of the faculties in the training of the noble one.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“Etassa, bhagavā, kālo; etassa, sugata, kālo}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Now is the time, Blessed One! Now is the time, Holy One.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yaṃ bhagavā ariyassa vinaye anuttaraṃ indriyabhāvanaṃ deseyya. Bhagavato sutvā bhikkhū dhāressantī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Let the Buddha teach the supreme development of the faculties in the training of the noble one. The mendicants will listen and remember it.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“Tenahānanda, suṇāhi, sādhukaṃ manasi karohi; bhāsissāmī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Well then, Ānanda, listen and pay close attention, I will speak.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“Evaṃ, bhante”ti kho āyasmā ānando bhagavato paccassosi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Yes, sir,” Ānanda replied.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Bhagavā etadavoca:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{The Buddha said this:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\vskip 0.05in
\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“Kathañcānanda, ariyassa vinaye anuttarā indriyabhāvanā hoti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“And how, Ānanda, is there the supreme development of the faculties in the training of the noble one?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Idhānanda, bhikkhuno cakkhunā rūpaṃ disvā uppajjati manāpaṃ, uppajjati amanāpaṃ, uppajjati manāpāmanāpaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{When a mendicant sees a sight with their eyes, liking, disliking, and both liking and disliking come up in them.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{So evaṃ pajānāti:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{They understand:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘uppannaṃ kho me idaṃ manāpaṃ, uppannaṃ amanāpaṃ, uppannaṃ manāpāmanāpaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘Liking, disliking, and both liking and disliking have come up in me.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Tañca kho saṅkhataṃ oḷārikaṃ paṭiccasamuppannaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{That’s conditioned, coarse, and dependently originated.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Etaṃ santaṃ etaṃ paṇītaṃ yadidaṃ—}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{But this is peaceful and sublime, namely}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{upekkhā’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{equanimity.’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Tassa taṃ uppannaṃ manāpaṃ uppannaṃ amanāpaṃ uppannaṃ manāpāmanāpaṃ nirujjhati;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then the liking, disliking, and both liking and disliking that came up in them cease,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{upekkhā saṇṭhāti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{and equanimity becomes stabilized.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Seyyathāpi, ānanda, cakkhumā puriso ummīletvā vā nimīleyya, nimīletvā vā ummīleyya;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It’s like how a person with good sight might open their eyes then shut them; or might shut their eyes then open them.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evameva kho, ānanda, yassa kassaci evaṃsīghaṃ evaṃtuvaṭaṃ evaṃappakasirena uppannaṃ manāpaṃ uppannaṃ amanāpaṃ uppannaṃ manāpāmanāpaṃ nirujjhati, upekkhā saṇṭhāti—}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Such is the speed, the swiftness, the ease with which any liking, disliking, and both liking and disliking at all that came up in them cease, and equanimity becomes stabilized.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ayaṃ vuccatānanda, ariyassa vinaye anuttarā indriyabhāvanā cakkhuviññeyyesu rūpesu.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{In the training of the noble one this is called the supreme development of the faculties regarding sights known by the eye.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\vskip 0.05in
\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{Puna caparaṃ, ānanda, bhikkhuno sotena saddaṃ sutvā uppajjati manāpaṃ, uppajjati amanāpaṃ, uppajjati manāpāmanāpaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}

\end{absolutelynopagebreak}

\setstretch{.7}
