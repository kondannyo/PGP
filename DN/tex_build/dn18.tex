\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{More than ninety devotees in Nādika have passed away having ended three fetters, and weakened greed, hate, and delusion. They’re once-returners, who will come back to this world once only, then make an end of suffering.}}\\
\end{addmargin}

\begin{addmargin}[1em]{2em}
