\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{They’re called an uneducated ordinary person who is bound to form, inside and out. They see neither the near shore nor the far shore. They’re born in bonds and die in bonds, and in bonds they go from this world to the next.}}\\
\end{addmargin}
