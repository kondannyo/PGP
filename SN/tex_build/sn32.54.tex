\setstretch{.7}

\setstretch{.5}
{\PaliGlossB{‘Why don’t we revel in our own kind of enjoyment?’ Then, in accordance with their wish, it becomes warm.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

{\PaliGlossB{This is the cause, this is the reason why sometimes it becomes warm.”}}\\
