\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{130. hāliddikānisutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{130. With Hāliddikāni}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ekaṃ samayaṃ āyasmā mahākaccāno avantīsu viharati kuraraghare papāte pabbate.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{At one time Venerable Mahākaccāna was staying in the land of the Avantis near Kuraraghara on Steep Mountain.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho hāliddikāni gahapati yenāyasmā mahākaccāno tenupasaṅkami … pe … ekamantaṃ nisinno kho hāliddikāni gahapati āyasmantaṃ mahākaccānaṃ etadavoca:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then the householder Hāliddikāni went up to Venerable Mahākaccāna … and asked him,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“vuttamidaṃ, bhante, bhagavatā:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Sir, this was said by the Buddha:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘dhātunānattaṃ paṭicca uppajjati phassanānattaṃ; phassanānattaṃ paṭicca uppajjati vedanānānattan’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘Diversity of elements gives rise to diversity of contacts, and diversity of contacts gives rise to diversity of feelings.’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kathaṃ nu kho, bhante, dhātunānattaṃ paṭicca uppajjati phassanānattaṃ; phassanānattaṃ paṭicca uppajjati vedanānānattan”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{How does diversity of elements give rise to diversity of contacts, and diversity of contacts gives rise to diversity of feelings?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“idha, gahapati, bhikkhu cakkhunā rūpaṃ disvā ‘manāpaṃ itthetan’ti pajānāti cakkhuviññāṇaṃ sukhavedaniyañca.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Householder, it’s when a mendicant sees a sight and understands it to be agreeable.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{phassaṃ paṭicca uppajjati sukhā vedanā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{There is eye consciousness; and pleasant feeling arises dependent on a contact to be experienced as pleasant.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{cakkhunā kho paneva rūpaṃ disvā ‘amanāpaṃ itthetan’ti pajānāti cakkhuviññāṇaṃ dukkhavedaniyañca.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then they see a sight and understand it to be disagreeable.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{phassaṃ paṭicca uppajjati dukkhā vedanā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{There is eye consciousness; and painful feeling arises dependent on a contact to be experienced as painful.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{cakkhunā kho paneva rūpaṃ disvā ‘upekkhāṭṭhāniyaṃ itthetan’ti pajānāti cakkhuviññāṇaṃ adukkhamasukhavedaniyañca.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then they see a sight and understand it to be a basis for equanimity.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{phassaṃ paṭicca uppajjati adukkhamasukhā vedanā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{There is eye consciousness; and neutral feeling arises dependent on a contact to be experienced as neutral.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{puna caparaṃ, gahapati, bhikkhu sotena saddaṃ sutvā … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Furthermore, a mendicant hears a sound with the ear …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ghānena gandhaṃ ghāyitvā … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{smells an odor with the nose …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{jivhāya rasaṃ sāyitvā … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{tastes a flavor with the tongue …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kāyena phoṭṭhabbaṃ phusitvā … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{feels a touch with the body …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{manasā dhammaṃ viññāya ‘manāpaṃ itthetan’ti pajānāti manoviññāṇaṃ sukhavedaniyañca.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{knows a thought with the mind and understands it to be agreeable.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{phassaṃ paṭicca uppajjati sukhā vedanā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{There is mind consciousness; and pleasant feeling arises dependent on a contact to be experienced as pleasant.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{manasā kho paneva dhammaṃ viññāya ‘amanāpaṃ itthetan’ti pajānāti manoviññāṇaṃ dukkhavedaniyañca.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then they know a thought and understand it to be disagreeable.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{phassaṃ paṭicca uppajjati dukkhā vedanā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{There is mind consciousness; and painful feeling arises dependent on a contact to be experienced as painful.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{manasā kho paneva dhammaṃ viññāya ‘upekkhāṭṭhāniyaṃ itthetan’ti pajānāti manoviññāṇaṃ adukkhamasukhavedaniyañca.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then they know a thought and understand it to be a basis for equanimity.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{phassaṃ paṭicca uppajjati adukkhamasukhā vedanā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Neutral feeling arises dependent on a contact to be experienced as neutral.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evaṃ kho, gahapati, dhātunānattaṃ paṭicca uppajjati phassanānattaṃ; phassanānattaṃ paṭicca uppajjati vedanānānattan”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{That’s how diversity of elements gives rise to diversity of contacts, and diversity of contacts gives rise to diversity of feelings.”}}\\
