\begin{addmargin}[1em]{2em}
\setstretch{.5}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{What are the things that are prone to being fettered? And what is the fetter?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
