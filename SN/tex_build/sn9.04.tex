\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{1. vanavagga}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{1. In the Woods}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{4. sambahulasutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{4. Several Mendicants Set Out Wandering}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ekaṃ samayaṃ sambahulā bhikkhū kosalesu viharanti aññatarasmiṃ vanasaṇḍe.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{At one time several mendicants were staying in the land of the Kosalans in a certain forest grove.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho te bhikkhū vassaṃvuṭṭhā temāsaccayena cārikaṃ pakkamiṃsu.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then after completing the three months of the rainy season residence, those mendicants set out wandering.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho yā tasmiṃ vanasaṇḍe adhivatthā devatā te bhikkhū apassantī paridevamānā tāyaṃ velāyaṃ imaṃ gāthaṃ abhāsi:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Not seeing those mendicants, the deity haunting that forest cried. And on that occasion they recited this verse:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“arati viya mejja khāyati,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Seeing so many vacated seats today,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{bahuke disvāna vivitte āsane;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{it seems to me that they must have become dissatisfied.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{te cittakathā bahussutā,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{They were so learned, such brilliant speakers!}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kome gotamasāvakā gatā”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Where have these disciples of Gotama gone?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evaṃ vutte, aññatarā devatā taṃ devataṃ gāthāya paccabhāsi:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{When they had spoken, another deity replied with this verse:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“māgadhaṃ gatā kosalaṃ gatā,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“They’ve gone to Magadha, they’ve gone to Kosala,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ekacciyā pana vajjibhūmiyā;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{and some are in the Vajjian lands.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{magā viya asaṅgacārino,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Like deer that wander free of ties,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{aniketā viharanti bhikkhavo”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{the mendicants live with no abode.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}
