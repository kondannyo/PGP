\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘appakā te sattā lokasmiṃ ye uḷāre uḷāre bhoge labhitvā na ceva majjanti, na ca pamajjanti, na ca kāmesu gedhaṃ āpajjanti, na ca sattesu vippaṭipajjanti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}

\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho eteva bahutarā sattā lokasmiṃ ye uḷāre uḷāre bhoge labhitvā majjanti ceva pamajjanti, ca kāmesu ca gedhaṃ āpajjanti, sattesu ca vippaṭipajjantī’”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{There are many more who, when they obtain luxury possessions, do get indulgent and negligent, giving in to greed for sensual pleasures, and doing the wrong thing by others.’”}}\\
\end{addmargin}
\end{absolutelynopagebreak}
