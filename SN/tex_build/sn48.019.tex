\begin{addmargin}[1em]{2em}
\setstretch{.5}

\end{addmargin}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{How is someone accomplished regarding the faculties defined?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}

{\PaliGlossA{“idha, bhikkhu, bhikkhu saddhindriyaṃ bhāveti upasamagāmiṃ sambodhagāmiṃ, vīriyindriyaṃ bhāveti upasamagāmiṃ sambodhagāmiṃ, satindriyaṃ bhāveti upasamagāmiṃ sambodhagāmiṃ, samādhindriyaṃ bhāveti upasamagāmiṃ sambodhagāmiṃ, paññindriyaṃ bhāveti upasamagāmiṃ sambodhagāmiṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicant, it’s when a mendicant develops the faculties of faith, energy, mindfulness, immersion, and wisdom that lead to peace and awakening.}}\\
\end{addmargin}
