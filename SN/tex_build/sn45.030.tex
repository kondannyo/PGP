\begin{absolutelynopagebreak}

\setstretch{.5}
{\PaliGlossB{Then Venerable Uttiya went up to the Buddha … and asked him,}}\\

\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}

{\PaliGlossB{“Just now, sir, as I was in private retreat this thought came to mind.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘The Buddha has spoken of the five kinds of sensual stimulation.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}

\begin{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{I have spoken of these five kinds of sensual stimulation.}}\\

\begin{absolutelynopagebreak}

\setstretch{.5}

\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sotaviññeyyā saddā … pe …}}\\

\setstretch{.5}
{\PaliGlossB{Sounds known by the ear …}}\\

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ghānaviññeyyā gandhā … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Smells known by the nose …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{jivhāviññeyyā rasā … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Tastes known by the tongue …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kāyaviññeyyā phoṭṭhabbā iṭṭhā kantā manāpā piyarūpā kāmūpasaṃhitā rajanīyā—}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Touches known by the body that are likable, desirable, agreeable, pleasant, sensual, and arousing.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ime kho, uttiya, pañca kāmaguṇā vuttā mayā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{These are the five kinds of sensual stimulation that I’ve spoken of.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{imesaṃ kho, uttiya, pañcannaṃ kāmaguṇānaṃ pahānāya ariyo aṭṭhaṅgiko maggo bhāvetabbo.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{The noble eightfold path should be developed to give up these five kinds of sensual stimulation.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{katamo ariyo aṭṭhaṅgiko maggo?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{What is the noble eightfold path?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}

{\PaliGlossB{It is right view, right thought, right speech, right action, right livelihood, right effort, right mindfulness, and right immersion.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{imesaṃ kho, uttiya, pañcannaṃ kāmaguṇānaṃ pahānāya ayaṃ ariyo aṭṭhaṅgiko maggo bhāvetabbo”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{This is the noble eightfold path that should be developed to give up these five kinds of sensual stimulation.”}}\\
\end{addmargin}
