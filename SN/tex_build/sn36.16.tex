\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho āyasmā ānando yena bhagavā tenupasaṅkami; upasaṅkamitvā bhagavantaṃ abhivādetvā ekamantaṃ nisīdi. ekamantaṃ nisinnaṃ kho āyasmantaṃ ānandaṃ bhagavā etadavoca:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then Venerable Ānanda went up to the Buddha, bowed, and sat down to one side. The Buddha said to him,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“katamā nu kho, ānanda, vedanā, katamo vedanāsamudayo, katamo vedanānirodho, katamā vedanānirodhagāminī paṭipadā?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Ānanda, what is feeling? What’s the origin of feeling? What’s the cessation of feeling? What’s the practice that leads to the cessation of feeling?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ko vedanāya assādo, ko ādīnavo, kiṃ nissaraṇan”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{And what is feeling’s gratification, drawback, and escape?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“bhagavaṃmūlakā no, bhante, dhammā bhagavaṃnettikā bhagavaṃpaṭisaraṇā. sādhu, bhante, bhagavantaññeva paṭibhātu etassa bhāsitassa attho. bhagavato sutvā bhikkhū dhāressantī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Our teachings are rooted in the Buddha. He is our guide and our refuge. Sir, may the Buddha himself please clarify the meaning of this. The mendicants will listen and remember it.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“tena hi, ānanda, suṇohi, sādhukaṃ manasi karohi; bhāsissāmī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Well then, Ānanda, listen and pay close attention, I will speak.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“evaṃ, bhante”ti kho āyasmā ānando bhagavato paccassosi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Yes, sir,” Ānanda replied.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}

\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
\setstretch{.5}

\end{addmargin}
\end{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{pleasant, painful, and neutral.}}\\
\end{addmargin}
\end{absolutelynopagebreak}
