\begin{addmargin}[1em]{2em}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘In my own way I should practice diligence, mindfulness, and protecting the mind regarding the five kinds of sensual stimulation that I formerly experienced—which have passed, ceased, and perished.’}}\\

\begin{addmargin}[1em]{2em}
\setstretch{.5}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{So in your own way you should practice diligence, mindfulness, and protecting the mind regarding the five kinds of sensual stimulation that I formerly experienced—which have passed, ceased, and perished.}}\\
