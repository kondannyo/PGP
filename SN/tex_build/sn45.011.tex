\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sāvatthinidānaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{At Sāvatthī.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“icchāmahaṃ, bhikkhave, aḍḍhamāsaṃ paṭisallīyituṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicants, I wish to go on retreat for a fortnight.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{namhi kenaci upasaṅkamitabbo, aññatra ekena piṇḍapātanīhārakenā”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{No-one should approach me, except for the one who brings my alms-food.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“evaṃ, bhante”ti kho te bhikkhū bhagavato paṭissutvā nāssudha koci bhagavantaṃ upasaṅkamati, aññatra ekena piṇḍapātanīhārakena.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Yes, sir,” replied those mendicants. And no-one approached him, except for the one who brought the alms-food.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho bhagavā tassa aḍḍhamāsassa accayena paṭisallānā vuṭṭhito bhikkhū āmantesi:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then after a fortnight had passed, the Buddha came out of retreat and addressed the mendicants:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“yena svāhaṃ, bhikkhave, vihārena paṭhamābhisambuddho viharāmi, tassa padesena vihāsiṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicants, I’ve been practicing part of the meditation I practiced when I was first awakened.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{so evaṃ pajānāmi:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{I understand that}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘micchādiṭṭhipaccayāpi vedayitaṃ;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{there’s feeling conditioned by wrong view}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sammādiṭṭhipaccayāpi vedayitaṃ … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{and feeling conditioned by right view. …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{micchāsamādhipaccayāpi vedayitaṃ;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{There’s feeling conditioned by wrong immersion,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sammāsamādhipaccayāpi vedayitaṃ;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{and feeling conditioned by right immersion.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{chandapaccayāpi vedayitaṃ;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{There’s feeling conditioned by desire,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{vitakkapaccayāpi vedayitaṃ;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{by thought,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{saññāpaccayāpi vedayitaṃ;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{and by perception.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{chando ca avūpasanto hoti, vitakko ca avūpasanto hoti, saññā ca avūpasantā hoti, tappaccayāpi vedayitaṃ;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{As long as desire, thought, and perception are not stilled, there is feeling conditioned by that.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{chando ca vūpasanto hoti, vitakko ca vūpasanto hoti, saññā ca vūpasantā hoti, tappaccayāpi vedayitaṃ;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{When desire, thought, and perception are stilled, there is feeling conditioned by that.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{appattassa pattiyā atthi āyāmaṃ, tasmimpi ṭhāne anuppatte tappaccayāpi vedayitan’”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{There is effort to attain the unattained. When that state has been attained, there is also feeling conditioned by that.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}
