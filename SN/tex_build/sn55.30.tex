\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{30. nandakalicchavisutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{30. With Nandaka the Licchavi}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ekaṃ samayaṃ bhagavā vesāliyaṃ viharati mahāvane kūṭāgārasālāyaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{At one time the Buddha was staying near Vesālī, at the Great Wood, in the hall with the peaked roof.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho nandako licchavimahāmatto yena bhagavā tenupasaṅkami; upasaṅkamitvā bhagavantaṃ abhivādetvā ekamantaṃ nisīdi. ekamantaṃ nisinnaṃ kho nandakaṃ licchavimahāmattaṃ bhagavā etadavoca:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then Nandaka the Licchavi minister went up to the Buddha, bowed, and sat down to one side. The Buddha said to him:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“catūhi kho, nandaka, dhammehi samannāgato ariyasāvako sotāpanno hoti avinipātadhammo niyato sambodhiparāyaṇo.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Nandaka, a noble disciple who has four things is a stream-enterer, not liable to be reborn in the underworld, bound for awakening.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{katamehi catūhi?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{What four?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{idha, nandaka, ariyasāvako buddhe aveccappasādena samannāgato hoti—}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It’s when a noble disciple has experiential confidence in the Buddha …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{itipi so bhagavā … pe … satthā devamanussānaṃ buddho bhagavāti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{    -}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{dhamme … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{the teaching …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{saṅghe … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{the Saṅgha …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ariyakantehi sīlehi samannāgato hoti akhaṇḍehi … pe … samādhisaṃvattanikehi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{And they have the ethical conduct loved by the noble ones … leading to immersion.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{imehi kho, nandaka, catūhi dhammehi samannāgato ariyasāvako sotāpanno hoti avinipātadhammo niyato sambodhiparāyaṇo.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{A noble disciple who has these four things is a stream-enterer, not liable to be reborn in the underworld, bound for awakening.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{imehi ca pana, nandaka, catūhi dhammehi samannāgato ariyasāvako āyunā saṃyutto hoti dibbenapi mānusenapi;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{A noble disciple who has these four things is guaranteed long life,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{vaṇṇena saṃyutto hoti dibbenapi mānusenapi;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{beauty,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sukhena saṃyutto hoti dibbenapi mānusenapi;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{happiness,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yasena saṃyutto hoti dibbenapi mānusenapi;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{fame,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ādhipateyyena saṃyutto hoti dibbenapi mānusenapi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{and sovereignty, both human and divine.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{taṃ kho panāhaṃ, nandaka, nāññassa samaṇassa vā brāhmaṇassa vā sutvā vadāmi. api ca yadeva mayā sāmaṃ ñātaṃ sāmaṃ diṭṭhaṃ sāmaṃ viditaṃ, tadevāhaṃ vadāmī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Now, I don’t say this because I’ve heard it from some other ascetic or brahmin. I only say it because I’ve known, seen, and realized it for myself.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evaṃ vutte, aññataro puriso nandakaṃ licchavimahāmattaṃ etadavoca:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{When he had spoken, a certain person said to Nandaka:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“nahānakālo, bhante”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Sir, it is time to bathe.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“alaṃ dāni, bhaṇe, etena bāhirena nahānena.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Enough now, my man, with that exterior bath.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{alamidaṃ ajjhattaṃ nahānaṃ bhavissati, yadidaṃ—}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{This interior bathing will do for me, that is,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{bhagavati pasādo”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{confidence in the Buddha.”}}\\

\end{absolutelynopagebreak}
