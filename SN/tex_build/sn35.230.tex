\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{230. bāḷisikopamasutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{230. The Simile of the Fisherman}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“seyyathāpi, bhikkhave, bāḷisiko āmisagatabaḷisaṃ gambhīre udakarahade pakkhipeyya.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicants, suppose a fisherman was to cast a baited hook into a deep lake.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tamenaṃ aññataro āmisacakkhu maccho gileyya.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Seeing the bait, a fish would swallow it.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evañhi so, bhikkhave, maccho gilitabaḷiso bāḷisikassa anayaṃ āpanno byasanaṃ āpanno yathākāmakaraṇīyo bāḷisikassa.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{And so the fish that swallowed the hook would meet with tragedy and disaster, and the fisherman can do what he wants with it.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evameva kho, bhikkhave, chayime baḷisā lokasmiṃ anayāya sattānaṃ vadhāya pāṇinaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{In the same way, there are these six hooks in the world that mean tragedy and slaughter for living creatures.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{katame cha?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{What six?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{santi, bhikkhave, cakkhuviññeyyā rūpā iṭṭhā kantā manāpā piyarūpā kāmūpasaṃhitā rajanīyā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{There are sights known by the eye that are likable, desirable, agreeable, pleasant, sensual, and arousing.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tañce, bhikkhu, abhinandati abhivadati ajjhosāya tiṭṭhati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{If a mendicant approves, welcomes, and keeps clinging to them,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ayaṃ vuccati, bhikkhave, bhikkhu gilitabaḷiso, mārassa anayaṃ āpanno byasanaṃ āpanno yathākāmakaraṇīyo pāpimato … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{they’re called a mendicant who has swallowed Māra’s hook. They’ve met with tragedy and disaster, and the Wicked One can do what he wants with them.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{santi, bhikkhave, jivhāviññeyyā rasā … pe ….}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{There are sounds … smells … tastes … touches …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{santi, bhikkhave, manoviññeyyā dhammā iṭṭhā kantā manāpā piyarūpā kāmūpasaṃhitā rajanīyā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{thoughts known by the mind that are likable, desirable, agreeable, pleasant, sensual, and arousing.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tañce, bhikkhu, abhinandati abhivadati ajjhosāya tiṭṭhati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{If a mendicant approves, welcomes, and keeps clinging to them,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ayaṃ vuccati, bhikkhave, bhikkhu gilitabaḷiso mārassa anayaṃ āpanno byasanaṃ āpanno yathākāmakaraṇīyo pāpimato.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{they’re called a mendicant who has swallowed Māra’s hook. They’ve met with tragedy and disaster, and the Wicked One can do what he wants with them.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{santi ca, bhikkhave, cakkhuviññeyyā rūpā iṭṭhā kantā manāpā piyarūpā kāmūpasaṃhitā rajanīyā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{There are sights known by the eye that are likable, desirable, agreeable, pleasant, sensual, and arousing.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tañce, bhikkhu, nābhinandati nābhivadati nājjhosāya tiṭṭhati. ayaṃ vuccati, bhikkhave, bhikkhu na gilitabaḷiso mārassa abhedi baḷisaṃ paribhedi baḷisaṃ na anayaṃ āpanno na byasanaṃ āpanno na yathākāmakaraṇīyo pāpimato … pe ….}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{If a mendicant doesn’t approve, welcome, and keep clinging to them, they’re called a mendicant who hasn’t swallowed Māra’s hook. They’ve broken the hook, destroyed it. They haven’t met with tragedy and disaster, and the Wicked One cannot do what he wants with them.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{santi, bhikkhave, jivhāviññeyyā rasā … pe ….}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{There are sounds … smells … tastes … touches …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{santi, bhikkhave, manoviññeyyā dhammā iṭṭhā kantā manāpā piyarūpā kāmūpasaṃhitā rajanīyā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{thoughts known by the mind that are likable, desirable, agreeable, pleasant, sensual, and arousing.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tañce bhikkhu nābhinandati nābhivadati nājjhosāya tiṭṭhati, ayaṃ vuccati, bhikkhave, bhikkhu na gilitabaḷiso mārassa abhedi baḷisaṃ paribhedi baḷisaṃ na anayaṃ āpanno na byasanaṃ āpanno na yathākāmakaraṇīyo pāpimato”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{If a mendicant doesn’t approve, welcome, and keep clinging to them, they’re called a mendicant who hasn’t swallowed Māra’s hook. They’ve broken the hook, destroyed it. They haven’t met with tragedy and disaster, and the Wicked One cannot do what he wants with them.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}
