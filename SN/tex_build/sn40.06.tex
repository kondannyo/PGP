\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{1. moggallānavagga}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{1. By Moggallāna}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{6. viññāṇañcāyatanapañhāsutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{6. A Question About the Dimension of Infinite Consciousness}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“‘viññāṇañcāyatanaṃ, viññāṇañcāyatanan’ti vuccati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“They speak of this thing called the ‘dimension of infinite consciousness’.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{katamaṃ nu kho viññāṇañcāyatananti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{What is the dimension of infinite consciousness?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tassa mayhaṃ, āvuso, etadahosi:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It occurred to me:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘idha bhikkhu sabbaso ākāsānañcāyatanaṃ samatikkamma anantaṃ viññāṇanti viññāṇañcāyatanaṃ upasampajja viharati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘It’s when a mendicant, going totally beyond the dimension of infinite space, aware that “consciousness is infinite”, enters and remains in the dimension of infinite consciousness.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{idaṃ vuccati viññāṇañcāyatanan’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{This is called the dimension of infinite consciousness.’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{so khvāhaṃ, āvuso, sabbaso ākāsānañcāyatanaṃ samatikkamma anantaṃ viññāṇanti viññāṇañcāyatanaṃ upasampajja viharāmi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{And so … I was entering and remaining in the dimension of infinite consciousness.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tassa mayhaṃ, āvuso, iminā vihārena viharato ākāsānañcāyatanasahagatā saññāmanasikārā samudācaranti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{While I was in that meditation, perceptions and attentions accompanied by the dimension of infinite space beset me.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho maṃ, āvuso, bhagavā iddhiyā upasaṅkamitvā etadavoca:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then the Buddha came up to me with his psychic power and said,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘moggallāna, moggallāna.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘Moggallāna, Moggallāna!}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{mā, brāhmaṇa, viññāṇañcāyatanaṃ pamādo, viññāṇañcāyatane cittaṃ saṇṭhapehi, viññāṇañcāyatane cittaṃ ekodiṃ karohi, viññāṇañcāyatane cittaṃ samādahā’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Don’t neglect the dimension of infinite consciousness, brahmin! Settle your mind in the dimension of infinite consciousness; unify your mind and immerse it in the dimension of infinite consciousness.’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{so khvāhaṃ, āvuso, aparena samayena sabbaso ākāsānañcāyatanaṃ samatikkamma anantaṃ viññāṇanti viññāṇañcāyatanaṃ upasampajja vihāsiṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{And so, after some time … I entered and remained in the dimension of infinite consciousness.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yañhi taṃ, āvuso, sammā vadamāno vadeyya … pe … mahābhiññataṃ patto”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{So if anyone should be rightly called a disciple who attained to great direct knowledge with help from the Teacher, it’s me.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}
