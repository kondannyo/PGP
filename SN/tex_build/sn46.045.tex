\begin{addmargin}[1em]{2em}
\setstretch{.5}

\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{How is a person who is wise, no idiot defined?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicant, they’re called wise, no idiot because they’ve developed and cultivated the seven awakening factors.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}

\setstretch{.5}
{\PaliGlossB{The awakening factors of mindfulness, investigation of principles, energy, rapture, tranquility, immersion, and equanimity.}}\\

\end{absolutelynopagebreak}
