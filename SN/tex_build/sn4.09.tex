\begin{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
\setstretch{.5}

\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tatra kho bhagavā bhikkhū āmantesi:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{There the Buddha addressed the mendicants,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\setstretch{.7}
{\PaliGlossA{“bhikkhavo”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicants!”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“bhadante”ti te bhikkhū bhagavato paccassosuṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Venerable sir,” they replied.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{bhagavā etadavoca:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{The Buddha said this:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“appamidaṃ, bhikkhave, manussānaṃ āyu.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicants, the life span of humans is short.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{gamanīyo samparāyo, kattabbaṃ kusalaṃ, caritabbaṃ brahmacariyaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{You must go to the next life. So you should do what is skillful, you should practice the spiritual life.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{natthi jātassa amaraṇaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{No-one born is immortal.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yo, bhikkhave, ciraṃ jīvati, so vassasataṃ appaṃ vā bhiyyo”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{A long life is a hundred years or a little more.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho māro pāpimā yena bhagavā tenupasaṅkami; upasaṅkamitvā bhagavantaṃ gāthāya ajjhabhāsi:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then Māra the Wicked went up to the Buddha and addressed him in verse:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“dīghamāyu manussānaṃ,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“The life of humans is long!}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{na naṃ hīḷe suporiso;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{A good person wouldn’t scorn it.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{careyya khīramattova,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Live like a suckling babe,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{natthi maccussa āgamo”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{for Death has not come for you.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“appamāyu manussānaṃ,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“The life of humans is short,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{hīḷeyya naṃ suporiso;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{and a good person scorns it.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{careyyādittasīsova,}}\\

\setstretch{.5}
{\PaliGlossB{They should live as though their head was on fire,}}\\

{\PaliGlossB{for Death comes for everyone.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}
