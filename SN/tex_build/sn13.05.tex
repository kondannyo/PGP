\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Which is more: the seven clay balls the size of jujube seeds, or the great earth?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Sir, the great earth is certainly more.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

{\PaliGlossB{The seven clay balls the size of jujube seeds are tiny.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Compared to the great earth, it’s not nearly a hundredth, a thousandth, or a hundred thousandth part.”}}\\
