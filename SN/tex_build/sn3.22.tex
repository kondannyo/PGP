\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{22. ayyikāsutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{22. Grandmother}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sāvatthinidānaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{At Sāvatthī.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ekamantaṃ nisinnaṃ kho rājānaṃ pasenadiṃ kosalaṃ bhagavā etadavoca:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{King Pasenadi of Kosala sat to one side, and the Buddha said to him,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“handa kuto nu tvaṃ, mahārāja, āgacchasi divādivassā”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“So, great king, where are you coming from in the middle of the day?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“ayyikā me, bhante, kālaṅkatā jiṇṇā vuḍḍhā mahallikā addhagatā vayoanuppattā vīsavassasatikā jātiyā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Sir, my grandmother has passed away. She was old, elderly and senior. She was advanced in years and had reached the final stage of life; she was a hundred and twenty years old.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ayyikā kho pana me, bhante, piyā hoti manāpā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{But I loved my grandmother; she was dear to me.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{hatthiratanena cepāhaṃ, bhante, labheyyaṃ ‘mā me ayyikā kālamakāsī’ti, hatthiratanampāhaṃ dadeyyaṃ:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{If by giving away the elephant treasure I could get my grandmother back, I’d do it.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘mā me ayyikā kālamakāsī’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{    -}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{assaratanena cepāhaṃ, bhante, labheyyaṃ ‘mā me ayyikā kālamakāsī’ti, assaratanampāhaṃ dadeyyaṃ:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{If by giving away the horse treasure I could get my grandmother back, I’d do it.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘mā me ayyikā kālamakāsī’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{    -}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{gāmavarena cepāhaṃ, bhante, labheyyaṃ ‘mā me ayyikā kālamakāsī’ti, gāmavarampāhaṃ dadeyyaṃ:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{If by giving away a prize village I could get my grandmother back, I’d do it.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘mā me ayyikā kālamakāsī’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{    -}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{janapadapadesena cepāhaṃ, bhante, labheyyaṃ ‘mā me ayyikā kālamakāsī’ti, janapadapadesampāhaṃ dadeyyaṃ:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{If by giving away the whole country I could get my grandmother back, I’d do it.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘mā me ayyikā kālamakāsī’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{    -}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘sabbe sattā, mahārāja, maraṇadhammā maraṇapariyosānā maraṇaṃ anatītā’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Great king, all sentient beings are liable to die. Death is their end; they’re not exempt from death.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘acchariyaṃ, bhante, abbhutaṃ, bhante.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“It’s incredible, sir, it’s amazing,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yāvasubhāsitamidaṃ, bhante, bhagavatā—}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{how well said this was by the Buddha:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sabbe sattā maraṇadhammā maraṇapariyosānā maraṇaṃ anatītā’”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘All sentient beings are liable to die. Death is their end; they’re not exempt from death.’”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“evametaṃ, mahārāja, evametaṃ, mahārāja.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“That’s so true, great king! That’s so true!}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sabbe sattā maraṇadhammā maraṇapariyosānā maraṇaṃ anatītā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{All sentient beings are liable to die. Death is their end; they’re not exempt from death.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{seyyathāpi, mahārāja, yāni kānici kumbhakārabhājanāni āmakāni ceva pakkāni ca sabbāni tāni bhedanadhammāni bhedanapariyosānāni bhedanaṃ anatītāni;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It’s like the vessels made by potters. Whatever kind they are, whether baked or unbaked, all of them are liable to break apart. Breaking is their end; they’re not exempt from breakage.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evameva kho, mahārāja, sabbe sattā maraṇadhammā maraṇapariyosānā maraṇaṃ anatītā”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{In the same way, all sentient beings are liable to die. Death is their end; they’re not exempt from death.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{idamavoca … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{That is what the Buddha said. …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“sabbe sattā marissanti,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“All beings will die,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{maraṇantañhi jīvitaṃ;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{for life ends with death.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yathākammaṃ gamissanti,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{They pass on according to their deeds,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{puññapāpaphalūpagā;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{reaping the fruits of good and bad.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{nirayaṃ pāpakammantā,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Those who do bad go to hell,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{puññakammā ca suggatiṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{and if you do good you go to heaven.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tasmā kareyya kalyāṇaṃ,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{That’s why you should do good,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{nicayaṃ samparāyikaṃ;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{investing in the future life.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{puññāni paralokasmiṃ,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{The good deeds of sentient beings}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{patiṭṭhā honti pāṇinan”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{support them in the next world.”}}\\
\end{addmargin}
