\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{123. sutavantasutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{123. Educated}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ekaṃ samayaṃ āyasmā ca sāriputto āyasmā ca mahākoṭṭhiko bārāṇasiyaṃ viharanti isipatane migadāye.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{At one time Venerable Sāriputta and Venerable Mahākoṭṭhita were staying near Benares, in the deer park at Isipatana.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho āyasmā mahākoṭṭhiko sāyanhasamayaṃ paṭisallānā vuṭṭhito yenāyasmā sāriputto tenupasaṅkami; upasaṅkamitvā … pe … etadavoca:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then in the late afternoon, Venerable Mahākoṭṭhita came out of retreat, went to Venerable Sāriputta, bowed, sat down to one side, and said:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“sutavatāvuso sāriputta, bhikkhunā katame dhammā yoniso manasi kātabbā”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Reverend Sāriputta, what things should an educated mendicant properly attend to?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“sutavatāvuso koṭṭhika, bhikkhunā pañcupādānakkhandhā aniccato … pe … anattato yoniso manasi kātabbā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“An educated mendicant should properly attend to these five grasping aggregates as impermanent … as not-self.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{katame pañca?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{What five?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{seyyathidaṃ—rūpupādānakkhandho … pe … viññāṇupādānakkhandho.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{That is, the grasping aggregates of form, feeling, perception, choices, and consciousness.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sutavatāvuso koṭṭhika, bhikkhunā ime pañcupādānakkhandhā aniccato … pe … anattato yoniso manasi kātabbā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{An educated mendicant should properly attend to these five grasping aggregates as impermanent … as not-self.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ṭhānaṃ kho panetaṃ, āvuso, vijjati—yaṃ sutavā bhikkhu ime pañcupādānakkhandhe aniccato … pe … anattato yoniso manasi karonto sotāpattiphalaṃ sacchikareyyā”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It’s possible that an educated mendicant who regards the five grasping aggregates in this way will realize the fruit of stream-entry.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“sotāpannena panāvuso sāriputta, bhikkhunā katame dhammā yoniso manasi kātabbā”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“But Reverend Sāriputta, what things should a mendicant stream-enterer properly attend to?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“sotāpannenapi kho āvuso koṭṭhika, bhikkhunā ime pañcupādānakkhandhā aniccato … pe … anattato yoniso manasi kātabbā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“A mendicant stream-enterer should properly attend to these five grasping aggregates as impermanent … as not-self.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ṭhānaṃ kho panetaṃ, āvuso, vijjati—yaṃ sotāpanno bhikkhu ime pañcupādānakkhandhe aniccato … pe … anattato yoniso manasi karonto sakadāgāmiphalaṃ … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It’s possible that a mendicant stream-enterer who regards the five grasping aggregates in this way will realize the fruit of once-return.” …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{anāgāmiphalaṃ … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“It’s possible that a mendicant once-returner who regards the five grasping aggregates in this way will realize the fruit of non-return.” …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{arahattaphalaṃ sacchikareyyā”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“It’s possible that a mendicant non-returner who regards the five grasping aggregates in this way will realize the fruit of perfection.” …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“arahatā panāvuso sāriputta, katame dhammā yoniso manasi kātabbā”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“But Reverend Sāriputta, what things should a perfected one properly attend to?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“arahatāpi khvāvuso koṭṭhika, ime pañcupādānakkhandhā aniccato dukkhato rogato gaṇḍato sallato aghato ābādhato parato palokato suññato anattato yoniso manasi kātabbā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Reverend Koṭṭhita, a perfected one should properly attend to the five grasping aggregates as impermanent, as suffering, as diseased, as an abscess, as a dart, as misery, as an affliction, as alien, as falling apart, as empty, as not-self.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{natthi, khvāvuso, arahato uttari karaṇīyaṃ, katassa vā paticayo;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{A perfected one has nothing more to do, and nothing that needs improvement.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{api ca kho ime dhammā bhāvitā bahulīkatā diṭṭhadhammasukhavihārāya ceva saṃvattanti satisampajaññāya cā”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Still, these things, when developed and cultivated, lead to blissful meditation in the present life, and also to mindfulness and situational awareness.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}
