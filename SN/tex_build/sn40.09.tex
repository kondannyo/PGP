\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{1. moggallānavagga}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{1. By Moggallāna}}\\

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{9. animittapañhāsutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{9. A Question About the Signless}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“‘animitto cetosamādhi, animitto cetosamādhī’ti vuccati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“They speak of this thing called the ‘signless immersion of the heart’.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{katamo nu kho animitto cetosamādhīti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{What is the signless immersion of the heart?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tassa mayhaṃ, āvuso, etadahosi:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It occurred to me:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘idha bhikkhu sabbanimittānaṃ amanasikārā animittaṃ cetosamādhiṃ upasampajja viharati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘It’s when a mendicant, not focusing on any signs, enters and remains in the signless immersion of the heart.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ayaṃ vuccati animitto cetosamādhī’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{This is called the signless immersion of the heart.’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{so khvāhaṃ, āvuso, sabbanimittānaṃ amanasikārā animittaṃ cetosamādhiṃ upasampajja viharāmi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{And so … I was entering and remaining in the signless immersion of the heart.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tassa mayhaṃ, āvuso, iminā vihārena viharato nimittānusāri viññāṇaṃ hoti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{While I was in that meditation, my consciousness followed after signs.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho maṃ, āvuso, bhagavā iddhiyā upasaṅkamitvā etadavoca:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then the Buddha came up to me with his psychic power and said,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘moggallāna, moggallāna.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘Moggallāna, Moggallāna!}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{mā, brāhmaṇa, animittaṃ cetosamādhiṃ pamādo, animitte cetosamādhismiṃ cittaṃ saṇṭhapehi, animitte cetosamādhismiṃ cittaṃ ekodiṃ karohi, animitte cetosamādhismiṃ cittaṃ samādahā’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Don’t neglect the signless immersion of the heart, brahmin! Settle your mind in the signless immersion of the heart; unify your mind and immerse it in the signless immersion of the heart.’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{so khvāhaṃ, āvuso, aparena samayena sabbanimittānaṃ amanasikārā animittaṃ cetosamādhiṃ upasampajja vihāsiṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{And so, after some time … I entered and remained in the signless immersion of the heart.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yañhi taṃ, āvuso, sammā vadamāno vadeyya:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{So if anyone should be rightly called}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘satthārānuggahito sāvako mahābhiññataṃ patto’ti, mamaṃ taṃ sammā vadamāno vadeyya:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{a disciple who attained to great direct knowledge with help from the Teacher, it’s me.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘satthārānuggahito sāvako mahābhiññataṃ patto’”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{    -}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{navamaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
