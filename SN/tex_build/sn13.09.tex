\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Which is more: the seven pebbles the size of mustard seeds, or the Himalayas, the king of mountains?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}

\setstretch{.5}
{\PaliGlossB{“Sir, the Himalayas, the king of mountains, is certainly more.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\end{addmargin}
\end{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
