\begin{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Attractiveness, wealth, relatives, children, and ethical behavior.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{na, bhikkhave, mātugāmo rūpabalahetu vā bhogabalahetu vā ñātibalahetu vā puttabalahetu vā kāyassa bhedā paraṃ maraṇā sugatiṃ saggaṃ lokaṃ upapajjati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It is not because of the powers of attractiveness, wealth, relatives, or children that females, when their body breaks up, after death, are reborn in a good place, a heavenly realm.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sīlabalahetu kho, bhikkhave, mātugāmo kāyassa bhedā paraṃ maraṇā sugatiṃ saggaṃ lokaṃ upapajjati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It is because of the power of ethical behavior that females, when their body breaks up, after death, are reborn in a good place, a heavenly realm.}}\\
\end{addmargin}
\end{absolutelynopagebreak}
