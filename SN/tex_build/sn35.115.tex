\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“santi, bhikkhave, cakkhuviññeyyā rūpā, iṭṭhā kantā manāpā piyarūpā kāmūpasaṃhitā rajanīyā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicants, there are sights known by the eye that are likable, desirable, agreeable, pleasant, sensual, and arousing.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tañce bhikkhu abhinandati abhivadati ajjhosāya tiṭṭhati—}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{If a mendicant approves, welcomes, and keeps clinging to them,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ayaṃ vuccati, bhikkhave, bhikkhu baddho cakkhuviññeyyesu rūpesu, āvāsagato mārassa, mārassa vasaṃ gato, paṭimukkassa mārapāso.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{they’re called a mendicant who is bound when it comes to sights known by the eye. They’re trapped in Māra’s lair, fallen under Māra’s sway, and caught in Māra’s snare.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{baddho so mārabandhanena yathākāmakaraṇīyo pāpimato … pe ….}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{They’re bound by Māra’s bonds, and the Wicked One can do what he wants with them.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}

\end{addmargin}
\end{absolutelynopagebreak}
