\begin{addmargin}[1em]{2em}

\end{absolutelynopagebreak}

\setstretch{.5}

\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then King Pasenadi of Kosala came downstairs from the stilt longhouse, went to the Buddha, bowed, sat down to one side, and told him what had happened.}}\\
\end{addmargin}
