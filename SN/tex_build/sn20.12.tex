\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“siyā kho, bhikkhave, tasmiṃ jarasiṅgāle yā kāci kataññutā kataveditā, na tveva idhekacce sakyaputtiyapaṭiññe siyā yā kāci kataññutā kataveditā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“There might be some gratitude and thankfulness in that old jackal, but there is none in a certain person here who claims to be a follower of the Sakyan son.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘We will be grateful and thankful.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{We won’t forget even a small thing done for us.’}}\\
\end{addmargin}
\end{absolutelynopagebreak}
