\setstretch{.5}
{\PaliGlossB{65. Samiddhi’s Question About Māra}}\\

\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ekaṃ samayaṃ bhagavā rājagahe viharati veḷuvane kalandakanivāpe.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{At one time the Buddha was staying near Rājagaha, in the Bamboo Grove, the squirrels’ feeding ground.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho āyasmā samiddhi yena bhagavā … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then Venerable Samiddhi went up to the Buddha …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{bhagavantaṃ etadavoca:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{and said to him:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“‘māro, māro’ti, bhante, vuccati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Sir, they speak of this thing called ‘Māra’.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kittāvatā nu kho, bhante, māro vā assa mārapaññatti vā”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{How do we define Māra or what is known as Māra?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“yattha kho, samiddhi, atthi cakkhu, atthi rūpā, atthi cakkhuviññāṇaṃ, atthi cakkhuviññāṇaviññātabbā dhammā, atthi tattha māro vā mārapaññatti vā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Samiddhi, where there is the eye, sights, eye consciousness, and phenomena to be known by eye consciousness, there is Māra or what is known as Māra.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atthi sotaṃ, atthi saddā, atthi sotaviññāṇaṃ, atthi sotaviññāṇaviññātabbā dhammā, atthi tattha māro vā mārapaññatti vā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Where there is the ear, sounds, ear consciousness, and phenomena to be known by ear consciousness, there is Māra or what is known as Māra.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atthi ghānaṃ, atthi gandhā, atthi ghānaviññāṇaṃ, atthi ghānaviññāṇaviññātabbā dhammā, atthi tattha māro vā mārapaññatti vā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Where there is the nose, smells, nose consciousness, and phenomena to be known by nose consciousness, there is Māra or what is known as Māra.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atthi jivhā, atthi rasā, atthi jivhāviññāṇaṃ, atthi jivhāviññāṇaviññātabbā dhammā, atthi tattha māro vā mārapaññatti vā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Where there is the tongue, tastes, tongue consciousness, and phenomena to be known by tongue consciousness, there is Māra or what is known as Māra.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atthi kāyo, atthi phoṭṭhabbā, atthi kāyaviññāṇaṃ, atthi kāyaviññāṇaviññātabbā dhammā, atthi tattha māro vā mārapaññatti vā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Where there is the body, touches, body consciousness, and phenomena to be known by body consciousness, there is Māra or what is known as Māra.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atthi mano, atthi dhammā, atthi manoviññāṇaṃ, atthi manoviññāṇaviññātabbā dhammā, atthi tattha māro vā mārapaññatti vā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Where there is the mind, thoughts, mind consciousness, and phenomena to be known by mind consciousness, there is Māra or what is known as Māra.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yattha ca kho, samiddhi, natthi cakkhu, natthi rūpā, natthi cakkhuviññāṇaṃ, natthi cakkhuviññāṇaviññātabbā dhammā, natthi tattha māro vā mārapaññatti vā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Where there is no eye, no sights, no eye consciousness, and no phenomena to be known by eye consciousness, there is no Māra or what is known as Māra.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
