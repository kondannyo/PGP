{\PaliGlossA{1. sāriputtavagga}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{1. With Sāriputta}}\\

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{10. sucimukhīsutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{10. With Sucimukhī}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ekaṃ samayaṃ āyasmā sāriputto rājagahe viharati veḷuvane kalandakanivāpe.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{At one time Venerable Sāriputta was staying near Rājagaha, in the Bamboo Grove, the squirrels’ feeding ground.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho āyasmā sāriputto pubbaṇhasamayaṃ nivāsetvā pattacīvaramādāya rājagahe piṇḍāya pāvisi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then he robed up in the morning and, taking his bowl and robe, entered Rājagaha for alms.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{rājagahe sapadānaṃ piṇḍāya caritvā taṃ piṇḍapātaṃ aññataraṃ kuṭṭamūlaṃ nissāya paribhuñjati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{After wandering indiscriminately for alms-food in Rājagaha, he ate his alms-food by a wall.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho sucimukhī paribbājikā yenāyasmā sāriputto tenupasaṅkami; upasaṅkamitvā āyasmantaṃ sāriputtaṃ etadavoca:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then the wanderer Sucimukhī went up to Venerable Sāriputta and said to him:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“kiṃ nu kho, samaṇa, adhomukho bhuñjasī”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Ascetic, do you eat facing downwards?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“na khvāhaṃ, bhagini, adhomukho bhuñjāmī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“No, sister.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“tena hi, samaṇa, ubbhamukho bhuñjasī”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Well then, do you eat facing upwards?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“na khvāhaṃ, bhagini, ubbhamukho bhuñjāmī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“No, sister.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“tena hi, samaṇa, disāmukho bhuñjasī”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Well then, do you eat facing the cardinal directions?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“na khvāhaṃ, bhagini, disāmukho bhuñjāmī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“No, sister.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“tena hi, samaṇa, vidisāmukho bhuñjasī”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Well then, do you eat facing the intermediate directions?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“na khvāhaṃ, bhagini, vidisāmukho bhuñjāmī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“No, sister.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“‘kiṃ nu, samaṇa, adhomukho bhuñjasī’ti iti puṭṭho samāno ‘na khvāhaṃ, bhagini, adhomukho bhuñjāmī’ti vadesi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“When asked if you eat facing all these directions, you answer ‘no, sister’.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘tena hi, samaṇa, ubbhamukho bhuñjasī’ti iti puṭṭho samāno ‘na khvāhaṃ, bhagini, ubbhamukho bhuñjāmī’ti vadesi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{    -}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘tena hi, samaṇa, disāmukho bhuñjasī’ti iti puṭṭho samāno ‘na khvāhaṃ, bhagini, disāmukho bhuñjāmī’ti vadesi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{    -}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘tena hi, samaṇa, vidisāmukho bhuñjasī’ti iti puṭṭho samāno ‘na khvāhaṃ, bhagini, vidisāmukho bhuñjāmī’ti vadesi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{    -}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kathañcarahi, samaṇa, bhuñjasī”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{How exactly do you eat, ascetic?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“ye hi keci, bhagini, samaṇabrāhmaṇā vatthuvijjātiracchānavijjāya micchājīvena jīvikaṃ kappenti, ime vuccanti, bhagini, samaṇabrāhmaṇā ‘adhomukhā bhuñjantī’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Sister, those ascetics and brahmins who earn a living by geomancy—an unworthy branch of knowledge, a wrong livelihood—are said to eat facing downwards.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ye hi keci, bhagini, samaṇabrāhmaṇā nakkhattavijjātiracchānavijjāya micchājīvena jīvikaṃ kappenti, ime vuccanti, bhagini, samaṇabrāhmaṇā ‘ubbhamukhā bhuñjantī’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Those ascetics and brahmins who earn a living by astrology—an unworthy branch of knowledge, a wrong livelihood—are said to eat facing upwards.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ye hi keci, bhagini, samaṇabrāhmaṇā dūteyyapahiṇagamanānuyogāya micchājīvena jīvikaṃ kappenti, ime vuccanti, bhagini, samaṇabrāhmaṇā ‘disāmukhā bhuñjantī’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Those ascetics and brahmins who earn a living by running errands and messages—a wrong livelihood—are said to eat facing the cardinal directions.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ye hi keci, bhagini, samaṇabrāhmaṇā aṅgavijjātiracchānavijjāya micchājīvena jīvikaṃ kappenti, ime vuccanti, bhagini, samaṇabrāhmaṇā ‘vidisāmukhā bhuñjantī’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Those ascetics and brahmins who earn a living by palmistry—an unworthy branch of knowledge, a wrong livelihood—are said to eat facing the intermediate directions.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{so khvāhaṃ, bhagini, na vatthuvijjātiracchānavijjāya micchājīvena jīvikaṃ kappemi, na nakkhattavijjātiracchānavijjāya micchājīvena jīvikaṃ kappemi, na dūteyyapahiṇagamanānuyogāya micchājīvena jīvikaṃ kappemi, na aṅgavijjātiracchānavijjāya micchājīvena jīvikaṃ kappemi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{I don’t earn a living by any of these means.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{dhammena bhikkhaṃ pariyesāmi;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{I seek alms in a principled manner,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{dhammena bhikkhaṃ pariyesitvā bhuñjāmī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{and I eat it in a principled manner.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho sucimukhī paribbājikā rājagahe rathiyāya rathiyaṃ, siṅghāṭakena siṅghāṭakaṃ upasaṅkamitvā evamārocesi:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then Sucimukhī the wanderer went around Rājagaha from street to street, from square to square, and announced:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“dhammikaṃ samaṇā sakyaputtiyā āhāraṃ āhārenti;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“The Sakyan ascetics eat food in a principled manner!}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{anavajjaṃ samaṇā sakyaputtiyā āhāraṃ āhārenti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{The Sakyan ascetics eat food blamelessly!}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{detha samaṇānaṃ sakyaputtiyānaṃ piṇḍan”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Give almsfood to the Sakyan ascetics!”}}\\
\end{addmargin}
\end{absolutelynopagebreak}
