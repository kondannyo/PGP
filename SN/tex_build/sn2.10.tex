\setstretch{.5}
{\PaliGlossB{“Homage to you, Buddha, hero!}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{vippamuttosi sabbadhi;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{You’re freed in every way.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sambādhapaṭipannosmi,}}\\

\setstretch{.5}

\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tassa me saraṇaṃ bhavā”ti.}}\\

\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}

{\PaliGlossB{Then the Buddha addressed Rāhu in verse concerning the Sun God:}}\\
\end{addmargin}

\setstretch{.7}
{\PaliGlossA{“tathāgataṃ arahantaṃ,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“The Sun God has gone for refuge}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sūriyo saraṇaṃ gato;}}\\
\begin{addmargin}[1em]{2em}

\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{rāhu sūriyaṃ pamuñcassu,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}

\end{addmargin}
\end{absolutelynopagebreak}

\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
\setstretch{.5}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{verocano maṇḍalī uggatejo;}}\\

\setstretch{.5}
{\PaliGlossB{the shining sun, circle of magnificent flame.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{mā rāhu gilī caramantalikkhe,}}\\
\begin{addmargin}[1em]{2em}

\end{addmargin}
\end{absolutelynopagebreak}

\setstretch{.5}
{\PaliGlossB{Vepacitti addressed him in verse:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\setstretch{.5}
{\PaliGlossB{“Why the rush?}}\\
