\begin{absolutelynopagebreak}

{\PaliGlossA{43. tatiyaabhisandasutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{43. Overflowing Merit (3rd)}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“cattārome, bhikkhave, puññābhisandā, kusalābhisandā, sukhassāhārā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicants, there are these four kinds of overflowing merit, overflowing goodness that nurture happiness.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{katame cattāro?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{What four?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{idha, bhikkhave, ariyasāvako buddhe aveccappasādena samannāgato hoti—}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It’s when a noble disciple has experiential confidence in the Buddha …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{itipi so bhagavā … pe … satthā devamanussānaṃ buddho bhagavāti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{    -}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ayaṃ paṭhamo puññābhisando, kusalābhisando, sukhassāhāro.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{    -}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{puna caparaṃ, bhikkhave, ariyasāvako dhamme … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{the teaching …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{saṅghe … pe ….}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{the Saṅgha …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{puna caparaṃ, bhikkhave, ariyasāvako paññavā hoti udayatthagāminiyā paññāya samannāgato ariyāya nibbedhikāya sammā dukkhakkhayagāminiyā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Furthermore, a noble disciple is wise. They have the wisdom of arising and passing away which is noble, penetrative, and leads to the complete ending of suffering.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ayaṃ catuttho puññābhisando kusalābhisando sukhassāhāro.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{This is the fourth kind of overflowing merit, overflowing goodness that nurtures happiness.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ime kho, bhikkhave, cattāro puññābhisandā kusalābhisandā sukhassāhārā.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{These are the four kinds of overflowing merit, overflowing goodness that nurture happiness.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{imehi kho, bhikkhave, catūhi puññābhisandehi kusalābhisandehi samannāgatassa ariyasāvakassa na sukaraṃ puññassa pamāṇaṃ gaṇetuṃ:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{When a noble disciple has these four kinds of overflowing merit and goodness, it’s not easy to measure how much merit they have by saying that}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘ettako puññābhisando, kusalābhisando, sukhassāhāro’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{this is the extent of their overflowing merit, overflowing goodness that nurtures happiness.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho asaṅkhyeyyo appameyyo mahāpuññakkhandhotveva saṅkhyaṃ gacchatī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{It’s simply reckoned as an incalculable, immeasurable, great mass of merit.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{idamavoca bhagavā … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{That is what the Buddha said.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{satthā:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then the Holy One, the Teacher, went on to say:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“yo puññakāmo kusale patiṭṭhito,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“One who desires merit, grounded in the skillful,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{bhāveti maggaṃ amatassa pattiyā;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{develops the eightfold path for realizing the deathless.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{so dhammasārādhigamo khaye rato,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Once they’ve reached the heart of the teaching, delighting in ending,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{they don’t tremble at the approach of the King of Death.”}}\\

\end{absolutelynopagebreak}
