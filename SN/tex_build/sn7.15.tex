\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{15. mānatthaddhasutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{15. Stuck-Up}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sāvatthinidānaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{At Sāvatthī.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tena kho pana samayena mānatthaddho nāma brāhmaṇo sāvatthiyaṃ paṭivasati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Now at that time a brahmin named Stuck-Up was residing in Sāvatthī.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{so neva mātaraṃ abhivādeti, na pitaraṃ abhivādeti, na ācariyaṃ abhivādeti, na jeṭṭhabhātaraṃ abhivādeti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{He didn’t bow to his mother or father, his teacher, or his oldest brother.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tena kho pana samayena bhagavā mahatiyā parisāya parivuto dhammaṃ deseti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Now, at that time the Buddha was teaching Dhamma, surrounded by a large assembly.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho mānatthaddhassa brāhmaṇassa etadahosi:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then Stuck-Up thought,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“ayaṃ kho samaṇo gotamo mahatiyā parisāya parivuto dhammaṃ deseti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“The ascetic Gotama is teaching Dhamma, surrounded by a large assembly.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yannūnāhaṃ yena samaṇo gotamo tenupasaṅkameyyaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Why don’t I approach him?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sace maṃ samaṇo gotamo ālapissati, ahampi taṃ ālapissāmi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{If he speaks to me, I’ll speak to him.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{no ce maṃ samaṇo gotamo ālapissati, ahampi nālapissāmī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{But if he doesn’t speak, neither will I.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho mānatthaddho brāhmaṇo yena bhagavā tenupasaṅkami; upasaṅkamitvā tuṇhībhūto ekamantaṃ aṭṭhāsi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then the brahmin Stuck-Up went up to the Buddha, and stood silently to one side.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho bhagavā taṃ nālapi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{But the Buddha didn’t speak to him.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho mānatthaddho brāhmaṇo:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then Stuck-Up thought,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“nāyaṃ samaṇo gotamo kiñci jānātī”ti tatova puna nivattitukāmo ahosi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“This ascetic Gotama knows nothing!” And he wanted to go back from there right away.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho bhagavā mānatthaddhassa brāhmaṇassa cetasā cetoparivitakkamaññāya mānatthaddhaṃ brāhmaṇaṃ gāthāya ajjhabhāsi:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then the Buddha, knowing what Stuck-Up was thinking, addressed him in verse:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“na mānaṃ brāhmaṇa sādhu,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“It’s not good to foster conceit}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atthikassīdha brāhmaṇa;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{if you want what’s good for you, brahmin.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yena atthena āgacchi,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{You should foster the goal}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tamevamanubrūhaye”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{which brought you here.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho mānatthaddho brāhmaṇo:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then Stuck-Up thought,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“cittaṃ me samaṇo gotamo jānātī”ti tattheva bhagavato pādesu sirasā nipatitvā bhagavato pādāni mukhena ca paricumbati pāṇīhi ca parisambāhati, nāmañca sāveti:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“The ascetic Gotama knows my mind!” He bowed with his head to the Buddha’s feet, caressing them and covering them with kisses, and pronounced his name:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“mānatthaddhāhaṃ, bho gotama, mānatthaddhāhaṃ, bho gotamā”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Master Gotama, I am Stuck-Up! I am Stuck-Up!”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho sā parisā abbhutavittajātā ahosi:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then that assembly was stunned:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“acchariyaṃ vata bho, abbhutaṃ vata bho.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“It’s incredible, it’s amazing!}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ayañhi mānatthaddho brāhmaṇo neva mātaraṃ abhivādeti, na pitaraṃ abhivādeti, na ācariyaṃ abhivādeti, na jeṭṭhabhātaraṃ abhivādeti;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{This brahmin Stuck-Up doesn’t bow to his mother or father, his teacher, or his oldest brother.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha ca pana samaṇe gotame evarūpaṃ paramanipaccakāraṃ karotī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Yet he shows such utmost devotion to the ascetic Gotama!”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho bhagavā mānatthaddhaṃ brāhmaṇaṃ etadavoca:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then the Buddha said to the brahmin Stuck-Up,}}\\
\end{addmargin}

\begin{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Enough, brahmin. Get up, and take your own seat.}}\\

\end{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}

{\PaliGlossB{Regarding these you should not be conceited.}}\\

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tesu assa sagāravo,}}\\

\setstretch{.5}
{\PaliGlossB{They are who you should respect.}}\\

\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tyassa apacitā assu;}}\\

\setstretch{.5}
{\PaliGlossB{They are who you should esteem.}}\\

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tyassu sādhu supūjitā.}}\\

\setstretch{.5}

\setstretch{.7}
{\PaliGlossA{arahante sītībhūte,}}\\

\setstretch{.5}
{\PaliGlossB{And when you’ve humbled conceit, and aren’t stuck-up,}}\\
