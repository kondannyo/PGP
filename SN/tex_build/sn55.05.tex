\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho āyasmā sāriputto yena bhagavā tenupasaṅkami; upasaṅkamitvā bhagavantaṃ abhivādetvā ekamantaṃ nisīdi. ekamantaṃ nisinnaṃ kho āyasmantaṃ sāriputtaṃ bhagavā etadavoca:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then Sāriputta went up to the Buddha, bowed, and sat down to one side. The Buddha said to him:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“‘sotāpattiyaṅgaṃ, sotāpattiyaṅgan’ti hidaṃ, sāriputta, vuccati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Sāriputta, they speak of a ‘factor of stream-entry’.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{katamaṃ nu kho, sāriputta, sotāpattiyaṅgan”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{What is a factor of stream-entry?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“sappurisasaṃsevo hi, bhante, sotāpattiyaṅgaṃ, saddhammassavanaṃ sotāpattiyaṅgaṃ, yonisomanasikāro sotāpattiyaṅgaṃ, dhammānudhammappaṭipatti sotāpattiyaṅgan”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Sir, the factors of stream-entry are associating with good people, listening to the true teaching, proper attention, and practicing in line with the teaching.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“sādhu sādhu, sāriputta.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Good, good, Sāriputta!}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sappurisasaṃsevo hi, sāriputta, sotāpattiyaṅgaṃ, saddhammassavanaṃ sotāpattiyaṅgaṃ, yonisomanasikāro sotāpattiyaṅgaṃ, dhammānudhammappaṭipatti sotāpattiyaṅgaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{For the factors of stream-entry are associating with good people, listening to the true teaching, proper attention, and practicing in line with the teaching.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“ayameva hi, bhante, ariyo aṭṭhaṅgiko maggo soto, seyyathidaṃ—}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Sir, the stream is simply this noble eightfold path, that is:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sammādiṭṭhi, sammāsaṅkappo, sammāvācā, sammākammanto, sammāājīvo, sammāvāyāmo, sammāsati, sammāsamādhī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{right view, right thought, right speech, right action, right livelihood, right effort, right mindfulness, and right immersion.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}
