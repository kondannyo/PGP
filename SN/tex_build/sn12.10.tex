\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“pubbeva me, bhikkhave, sambodhā anabhisambuddhassa bodhisattasseva sato etadahosi:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Mendicants, before my awakening—when I was still unawakened but intent on awakening—I thought:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘kicchaṃ vatāyaṃ loko āpanno jāyati ca jīyati ca mīyati ca cavati ca upapajjati ca.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘Alas, this world has fallen into trouble. It’s born, grows old, dies, passes away, and is reborn,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha ca panimassa dukkhassa nissaraṇaṃ nappajānāti jarāmaraṇassa.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{yet it doesn’t understand how to escape from this suffering, from old age and death.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kudāssu nāma imassa dukkhassa nissaraṇaṃ paññāyissati jarāmaraṇassā’ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Oh, when will an escape be found from this suffering, from old age and death?’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\end{addmargin}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘When what exists is there old age and death? What is a condition for old age and death?’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘When rebirth exists there’s old age and death. Rebirth is a condition for old age and death.’}}\\
