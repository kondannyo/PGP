{\PaliGlossA{16. dukkarasutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{16. Hard to Do}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“kiṃ nu kho, āvuso sāriputta, imasmiṃ dhammavinaye dukkaran”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Reverend Sāriputta, in this teaching and training, what is hard to do?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“pabbajjā kho, āvuso, imasmiṃ dhammavinaye dukkarā”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Going forth, reverend, is hard to do in this teaching and training.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“pabbajitena panāvuso, kiṃ dukkaran”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“But what’s hard to do for someone who has gone forth?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“pabbajitena kho, āvuso, abhirati dukkarā”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“When you’ve gone forth it’s hard to be satisfied.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“abhiratena panāvuso, kiṃ dukkaran”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“But what’s hard to do for someone who is satisfied?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“abhiratena kho, āvuso, dhammānudhammappaṭipatti dukkarā”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“When you’re satisfied, it’s hard to practice in line with the teaching.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“kīvaciraṃ panāvuso, dhammānudhammappaṭipanno bhikkhu arahaṃ assā”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“But if a mendicant practices in line with the teaching, will it take them long to become a perfected one?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“naciraṃ, āvuso”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Not long, reverend.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}
