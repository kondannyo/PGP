\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“‘paramassāsappatto, paramassāsappatto’ti, āvuso sāriputta, vuccati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Reverend Sāriputta, they speak of this thing called ‘gaining ultimate solace’.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kittāvatā nu kho, āvuso, paramassāsappatto hotī”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{At what point do you gain ultimate solace?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“yato kho, āvuso, bhikkhu channaṃ phassāyatanānaṃ samudayañca atthaṅgamañca assādañca ādīnavañca nissaraṇañca yathābhūtaṃ viditvā anupādāvimutto hoti, ettāvatā kho, āvuso, paramassāsappatto hotī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“When a mendicant is freed by not grasping after truly understanding the six fields of contacts’ origin, ending, gratification, drawback, and escape, at that point they’ve gained ultimate solace.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}
