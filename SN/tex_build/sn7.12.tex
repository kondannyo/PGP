\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then the Buddha robed up in the morning and, taking his bowl and robe, went to the home of the brahmin Udaya.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho udayo brāhmaṇo bhagavato pattaṃ odanena pūresi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then Udaya filled the Buddha’s bowl with rice.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{dutiyampi kho bhagavā pubbaṇhasamayaṃ nivāsetvā pattacīvaramādāya yena udayassa brāhmaṇassa nivesanaṃ tenupasaṅkami … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{The next day … and the day after that … Udaya filled the Buddha’s bowl with rice.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tatiyampi kho udayo brāhmaṇo bhagavato pattaṃ odanena pūretvā bhagavantaṃ etadavoca:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{But when he had filled the Buddha’s bowl for a third time, he said to the Buddha,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
\setstretch{.5}

\setstretch{.5}
{\PaliGlossB{“Again and again, they sow the seed;}}\\

\end{absolutelynopagebreak}

\setstretch{.7}
{\PaliGlossA{punappunaṃ vassati devarājā;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{again and again, the lord god sends rain;}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{punappunaṃ khettaṃ kasanti kassakā,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{again and again, farmers plough the field;}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{punappunaṃ dhaññamupeti raṭṭhaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{again and again, grain is produced for the nation.}}\\

\end{absolutelynopagebreak}

{\PaliGlossA{punappunaṃ yācakā yācayanti,}}\\

\setstretch{.7}
{\PaliGlossA{punappunaṃ dānapatī dadanti;}}\\

{\PaliGlossA{punappunaṃ dānapatī daditvā,}}\\

\end{addmargin}

\setstretch{.5}
