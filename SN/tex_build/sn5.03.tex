\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{3. kisāgotamīsutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{3. With Kisāgotamī}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sāvatthinidānaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{At Sāvatthī.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho kisāgotamī bhikkhunī pubbaṇhasamayaṃ nivāsetvā pattacīvaramādāya sāvatthiṃ piṇḍāya pāvisi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then the nun Kisāgotamī robed up in the morning and, taking her bowl and robe, entered Sāvatthī for alms.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sāvatthiyaṃ piṇḍāya caritvā pacchābhattaṃ piṇḍapātapaṭikkantā yena andhavanaṃ tenupasaṅkami, upasaṅkamitvā divāvihārāya.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{She wandered for alms in Sāvatthī. After the meal, on her return from alms-round, she went to the Dark Forest,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{andhavanaṃ ajjhogāhetvā aññatarasmiṃ rukkhamūle divāvihāraṃ nisīdi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{plunged deep into it, and sat at the root of a tree for the day’s meditation.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho māro pāpimā kisāgotamiyā bhikkhuniyā bhayaṃ chambhitattaṃ lomahaṃsaṃ uppādetukāmo samādhimhā cāvetukāmo yena kisāgotamī bhikkhunī tenupasaṅkami; upasaṅkamitvā kisāgotamiṃ bhikkhuniṃ gāthāya ajjhabhāsi:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then Māra the Wicked, wanting to make the nun Kisāgotamī feel fear, terror, and goosebumps, wanting to make her fall away from immersion, went up to her and addressed her in verse:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“kiṃ nu tvaṃ mataputtāva,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Why do you sit alone and cry}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ekamāsi rudammukhī;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{as if your children have died?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{vanamajjhagatā ekā,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{You’ve come to the woods all alone—}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{purisaṃ nu gavesasī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{you must be looking for a man!”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho kisāgotamiyā bhikkhuniyā etadahosi:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then the nun Kisāgotamī thought,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“ko nu khvāyaṃ manusso vā amanusso vā gāthaṃ bhāsatī”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Who’s speaking this verse, a human or a non-human?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho kisāgotamiyā bhikkhuniyā etadahosi:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then she thought,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“māro kho ayaṃ pāpimā mama bhayaṃ chambhitattaṃ lomahaṃsaṃ uppādetukāmo samādhimhā cāvetukāmo gāthaṃ bhāsatī”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“This is Māra the Wicked, wanting to make me feel fear, terror, and goosebumps, wanting to make me fall away from immersion!”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho kisāgotamī bhikkhunī “māro ayaṃ pāpimā” iti viditvā māraṃ pāpimantaṃ gāthāhi paccabhāsi:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then Kisāgotamī, knowing that this was Māra the Wicked, replied to him in verse:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“accantaṃ mataputtāmhi,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“I’ve got over the death of children,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{purisā etadantikā;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{and I’m finished with men.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{na socāmi na rodāmi,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{I don’t grieve or lament,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{na taṃ bhāyāmi āvuso.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{and I’m not afraid of you, sir!}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sabbattha vihatā nandī,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Relishing is destroyed in every respect,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tamokkhandho padālito;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{and the mass of darkness is shattered.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{jetvāna maccuno senaṃ,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{I’ve defeated the army of death,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{viharāmi anāsavā”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{and live without defilements.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho māro pāpimā “jānāti maṃ kisāgotamī bhikkhunī”ti dukkhī dummano tatthevantaradhāyīti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then Māra the Wicked, thinking, “The nun Kisāgotamī knows me!” miserable and sad, vanished right there.}}\\
\end{addmargin}
\end{absolutelynopagebreak}
