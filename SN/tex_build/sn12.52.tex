\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{52. upādānasutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{52. Grasping}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sāvatthiyaṃ viharati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{At Sāvatthī.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“upādāniyesu, bhikkhave, dhammesu assādānupassino viharato taṇhā pavaḍḍhati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“There are things that are prone to being grasped. When you concentrate on the gratification provided by these things, your craving grows.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{taṇhāpaccayā upādānaṃ;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Craving is a condition for grasping.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{upādānapaccayā bhavo;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Grasping is a condition for continued existence.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{bhavapaccayā jāti;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Continued existence is a condition for rebirth.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{jātipaccayā jarāmaraṇaṃ sokaparidevadukkhadomanassupāyāsā sambhavanti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Rebirth is a condition for old age and death, sorrow, lamentation, pain, sadness, and distress to come to be.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evametassa kevalassa dukkhakkhandhassa samudayo hoti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{That is how this entire mass of suffering originates.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{seyyathāpi, bhikkhave, dasannaṃ vā kaṭṭhavāhānaṃ vīsāya vā kaṭṭhavāhānaṃ tiṃsāya vā kaṭṭhavāhānaṃ cattārīsāya vā kaṭṭhavāhānaṃ mahāaggikkhandho jaleyya.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Suppose a bonfire was burning with ten, twenty, thirty, or forty loads of wood.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tatra puriso kālena kālaṃ sukkhāni ceva tiṇāni pakkhipeyya, sukkhāni ca gomayāni pakkhipeyya, sukkhāni ca kaṭṭhāni pakkhipeyya.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{And from time to time someone would toss in dry grass, cow dung, or wood.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evañhi so, bhikkhave, mahāaggikkhandho tadāhāro tadupādāno ciraṃ dīghamaddhānaṃ jaleyya.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Fueled and sustained by that, the bonfire would burn for a long time.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evameva kho, bhikkhave, upādāniyesu dhammesu assādānupassino viharato taṇhā pavaḍḍhati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{In the same way, there are things that are prone to being grasped. When you concentrate on the gratification provided by these things, your craving grows.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{taṇhāpaccayā upādānaṃ … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Craving is a condition for grasping. …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evametassa kevalassa dukkhakkhandhassa samudayo hoti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{That is how this entire mass of suffering originates.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{upādāniyesu, bhikkhave, dhammesu ādīnavānupassino viharato taṇhā nirujjhati.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{There are things that are prone to being grasped. When you concentrate on the drawbacks of these things, your craving ceases.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{taṇhānirodhā upādānanirodho;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{When craving ceases, grasping ceases.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{upādānanirodhā bhavanirodho;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{When grasping ceases, continued existence ceases.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{bhavanirodhā jātinirodho;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{When continued existence ceases, rebirth ceases.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{jātinirodhā jarāmaraṇaṃ sokaparidevadukkhadomanassupāyāsā nirujjhanti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{When rebirth ceases, old age and death, sorrow, lamentation, pain, sadness, and distress cease.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evametassa kevalassa dukkhakkhandhassa nirodho hoti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{That is how this entire mass of suffering ceases.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{seyyathāpi, bhikkhave, dasannaṃ vā kaṭṭhavāhānaṃ vīsāya vā tiṃsāya vā cattārīsāya vā kaṭṭhavāhānaṃ mahāaggikkhandho jaleyya;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Suppose a bonfire was burning with ten, twenty, thirty, or forty loads of wood.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{tatra puriso na kālena kālaṃ sukkhāni ceva tiṇāni pakkhipeyya, na sukkhāni ca gomayāni pakkhipeyya, na sukkhāni ca kaṭṭhāni pakkhipeyya.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{And no-one would toss in dry grass, cow dung, or wood from time to time.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evañhi so, bhikkhave, mahāaggikkhandho purimassa ca upādānassa pariyādānā aññassa ca anupahārā anāhāro nibbāyeyya.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{As the original fuel is used up and no more is added, the bonfire would be extinguished due to lack of fuel.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evameva kho, bhikkhave, upādāniyesu dhammesu ādīnavānupassino viharato taṇhā nirujjhati,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{In the same way, there are things that are prone to being grasped. When you concentrate on the drawbacks of these things, your craving ceases.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{taṇhānirodhā upādānanirodho … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{When craving ceases, grasping ceases. …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evametassa kevalassa dukkhakkhandhassa nirodho hotī”ti.}}\\

\setstretch{.5}
{\PaliGlossB{That is how this entire mass of suffering ceases.”}}\\
\end{addmargin}
