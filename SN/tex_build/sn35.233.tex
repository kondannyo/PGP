\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{233. kāmabhūsutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{233. With Kāmabhū}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ekaṃ samayaṃ āyasmā ca ānando āyasmā ca kāmabhū kosambiyaṃ viharanti ghositārāme.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{At one time the venerables Ānanda and Kāmabhū were staying near Kosambi, in Ghosita’s Monastery.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho āyasmā kāmabhū sāyanhasamayaṃ paṭisallānā vuṭṭhito yenāyasmā ānando tenupasaṅkami; upasaṅkamitvā āyasmatā ānandena saddhiṃ sammodi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then in the late afternoon, Venerable Kāmabhū came out of retreat, went to Venerable Ānanda, and exchanged greetings with him.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sammodanīyaṃ kathaṃ sāraṇīyaṃ vītisāretvā ekamantaṃ nisīdi. ekamantaṃ nisinno kho āyasmā kāmabhū āyasmantaṃ ānandaṃ etadavoca:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{When the greetings and polite conversation were over, he sat down to one side and said to Ānanda:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“kiṃ nu kho, āvuso ānanda, cakkhu rūpānaṃ saṃyojanaṃ, rūpā cakkhussa saṃyojanaṃ … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Reverend Ānanda, which is it? Is the eye the fetter of sights, or are sights the fetter of the eye?}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{jivhā rasānaṃ saṃyojanaṃ, rasā jivhāya saṃyojanaṃ … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Is the ear … nose … tongue … body …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{mano dhammānaṃ saṃyojanaṃ, dhammā manassa saṃyojanan”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{mind the fetter of thoughts, or are thoughts the fetter of the mind?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“na kho, āvuso kāmabhū, cakkhu rūpānaṃ saṃyojanaṃ, na rūpā cakkhussa saṃyojanaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Reverend Kāmabhū, the eye is not the fetter of sights, nor are sights the fetter of the eye.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yañca tattha tadubhayaṃ paṭicca uppajjati chandarāgo, taṃ tattha saṃyojanaṃ … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{The fetter there is the desire and greed that arises from the pair of them.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{na jivhā rasānaṃ saṃyojanaṃ, na rasā jivhāya saṃyojanaṃ … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{The ear … nose … tongue … body …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{na mano dhammānaṃ saṃyojanaṃ, na dhammā manassa saṃyojanaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{mind is not the fetter of thoughts, nor are thoughts the fetter of the mind.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yañca tattha tadubhayaṃ paṭicca uppajjati chandarāgo taṃ tattha saṃyojanaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{The fetter there is the desire and greed that arises from the pair of them.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{seyyathāpi, āvuso, kāḷo ca balībaddo odāto ca balībaddo ekena dāmena vā yottena vā saṃyuttā assu.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Suppose there was a black ox and a white ox yoked by a single harness or yoke.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yo nu kho evaṃ vadeyya:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Would it be right to say that}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘kāḷo balībaddo odātassa balībaddassa saṃyojanaṃ, odāto balībaddo kāḷassa balībaddassa saṃyojanan’ti, sammā nu kho so vadamāno vadeyyā”ti?}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{the black ox is the yoke of the white ox, or the white ox is the yoke of the black ox?”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“no hetaṃ, āvuso”.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“No, reverend.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“na kho, āvuso, kāḷo balībaddo odātassa balībaddassa saṃyojanaṃ, napi odāto balībaddo kāḷassa balībaddassa saṃyojanaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{The black ox is not the yoke of the white ox, nor is the white ox the yoke of the black ox.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{yena ca kho te ekena dāmena vā yottena vā saṃyuttā, taṃ tattha saṃyojanaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{The yoke there is the single harness or yoke that they’re yoked by.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{evameva kho, āvuso, na cakkhu rūpānaṃ saṃyojanaṃ, na rūpā cakkhussa saṃyojanaṃ … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“In the same way, the eye is not the fetter of sights, nor are sights the fetter of the eye.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{na jivhā … pe …}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{The ear … nose … tongue … body …}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{mind is not the fetter of thoughts, nor are thoughts the fetter of the mind.}}\\
\end{addmargin}
\end{absolutelynopagebreak}
