\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{pubbā koṭi na paññāyati avijjānīvaraṇānaṃ sattānaṃ taṇhāsaṃyojanānaṃ sandhāvataṃ saṃsarataṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{No first point is found of sentient beings roaming and transmigrating, hindered by ignorance and fettered by craving.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{hoti so, bhikkhave, samayo yaṃ mahāsamuddo ussussati visussati na bhavati;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{There comes a time when the ocean dries up and evaporates and is no more.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{na tvevāhaṃ, bhikkhave, avijjānīvaraṇānaṃ sattānaṃ taṇhāsaṃyojanānaṃ sandhāvataṃ saṃsarataṃ dukkhassa antakiriyaṃ vadāmi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{But still, I say, there is no making an end of suffering for sentient beings roaming and transmigrating, hindered by ignorance and fettered by craving.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}

\setstretch{.5}
{\PaliGlossB{There comes a time when Sineru the king of mountains is burned up and destroyed, and is no more.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{seyyathāpi, bhikkhave, sā gaddulabaddho daḷhe khīle vā thambhe vā upanibaddho tameva khīlaṃ vā thambhaṃ vā anuparidhāvati anuparivattati;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Suppose a dog on a leash was tethered to a strong post or pillar. It would just keep running and circling around that post or pillar.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}

\setstretch{.5}
