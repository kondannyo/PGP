\begin{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{The Buddha was stricken by harrowing pains; physical feelings that were painful, sharp, severe, acute, unpleasant, and disagreeable.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}

\end{addmargin}

\begin{absolutelynopagebreak}
\setstretch{.7}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{And then he spread out his outer robe folded in four and laid down in the lion’s posture—on the right side, placing one foot on top of the other—mindful and aware.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho sattasatā satullapakāyikā devatāyo abhikkantāya rattiyā abhikkantavaṇṇā kevalakappaṃ maddakucchiṃ obhāsetvā yena bhagavā tenupasaṅkamiṃsu; upasaṅkamitvā bhagavantaṃ abhivādetvā ekamantaṃ aṭṭhaṃsu.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}

\end{addmargin}
\end{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
