\setstretch{.5}
{\PaliGlossB{“No brahmin in the world is ever purified}}\\

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{loke sīlavāpi tapokaraṃ;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{even though he’s ethical and mortifies himself.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{vijjācaraṇasampanno,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{But one accomplished in knowledge and conduct}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{so sujjhati na aññā itarā pajā”ti.}}\\
\begin{addmargin}[1em]{2em}

\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“bahumpi palapaṃ jappaṃ,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Even one who mutters many invocations}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{na jaccā hoti brāhmaṇo;}}\\

\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{antokasambu saṅkiliṭṭho,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}

\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kuhanaṃ upanissito.}}\\

\setstretch{.5}
{\PaliGlossB{supporting themselves by fraud.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{khattiyo brāhmaṇo vesso,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Regardless of whether you’re an aristocrat,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{suddo caṇḍālapukkuso;}}\\

\setstretch{.5}

\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{āraddhavīriyo pahitatto,}}\\

\setstretch{.7}
{\PaliGlossA{niccaṃ daḷhaparakkamo;}}\\
