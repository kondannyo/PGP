\begin{absolutelynopagebreak}

\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{As long as I live, may I be free of anger, or should anger arise, may I quickly get rid of it.}}\\

\begin{addmargin}[1em]{2em}
