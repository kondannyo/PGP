\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{9. araññāyatanaisisutta}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{9. Hermits in the Wilderness}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sāvatthiyaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{At Sāvatthī.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{“bhūtapubbaṃ, bhikkhave, sambahulā isayo sīlavanto kalyāṇadhammā araññāyatane paṇṇakuṭīsu sammanti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{“Once upon a time, mendicants, several hermits who were ethical, of good character, settled in leaf huts in a wilderness region.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho, bhikkhave, sakko ca devānamindo vepacitti ca asurindo yena te isayo sīlavanto kalyāṇadhammā tenupasaṅkamiṃsu.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then Sakka, lord of gods, and Vepacitti, lord of demons, went to those hermits.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho, bhikkhave, vepacitti asurindo paṭaliyo upāhanā ārohitvā khaggaṃ olaggetvā chattena dhāriyamānena aggadvārena assamaṃ pavisitvā te isayo sīlavante kalyāṇadhamme apabyāmato karitvā atikkami.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Vepacitti put on his boots, strapped on his sword, and, carrying a sunshade, entered the hermitage through the main gate. He walked right past those hermits, keeping them at a distance.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho, bhikkhave, sakko devānamindo paṭaliyo upāhanā orohitvā khaggaṃ aññesaṃ datvā chattaṃ apanāmetvā dvāreneva assamaṃ pavisitvā te isayo sīlavante kalyāṇadhamme anuvātaṃ pañjaliko namassamāno aṭṭhāsi.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Sakka took off his boots, gave his sword to others, and, putting down his sunshade, entered the hermitage through a gate he happened upon. He stood downwind of those hermits, revering them with joined palms.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{atha kho, bhikkhave, te isayo sīlavanto kalyāṇadhammā sakkaṃ devānamindaṃ gāthāya ajjhabhāsiṃsu:}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{Then those hermits addressed Sakka in verse:}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘gandho isīnaṃ ciradikkhitānaṃ,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘When hermits have been long ordained,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kāyā cuto gacchati mālutena;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{the odor of their bodies goes with the wind.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{ito paṭikkamma sahassanetta,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{You’d better leave, O thousand-eyed!}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{gandho isīnaṃ asuci devarājā’ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{The odor of the hermits is unclean, king of gods.’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{‘gandho isīnaṃ ciradikkhitānaṃ,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{‘When hermits have been long ordained,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{kāyā cuto gacchatu mālutena;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{let the odor of their bodies go with the wind.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{sucitrapupphaṃ sirasmiṃva mālaṃ,}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{We yearn for this odor, sirs,}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{gandhaṃ etaṃ paṭikaṅkhāma bhante;}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{like a colorful crown of flowers.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliGlossA{na hettha devā paṭikūlasaññino’”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\PaliGlossB{The gods don’t see it as repulsive.’”}}\\
\end{addmargin}
\end{absolutelynopagebreak}
