\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{evameva bhoto gotamassa pāvacanaṁ apagatasākhāpalāsaṁ apagatatacapapaṭikaṁ apagatapheggukaṁ suddhaṁ, sāre patiṭṭhitaṁ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{so too, this discourse of Master Gotama’s is divested of branches and foliage, divested of bark and sapwood, and is pure, consisting entirely of heartwood.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\vskip 0.05in
\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{“abhikkantaṃ, bho gotama, abhikkantaṃ, bho gotama!}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{“Magnificent, Master Gotama!  Magnificent, Master Gotama!}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{seyyathāpi, bho gotama, nikkujjitaṃ vā ukkujjeyya, paṭicchannaṃ vā vivareyya, mūḷhassa vā maggaṃ ācikkheyya, andhakāre vā telapajjotaṃ dhāreyya — cakkhumanto rūpāni dakkhantīti; evamevaṃ bhotā gotamena anekapariyāyena dhammo pakāsito.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{Master Gotama has made the Dhamma clear in many ways, as though he were turning upright what had been overthrown, revealing what was hidden, showing the way to one who was lost, or holding up a lamp in the dark for those with eyesight to see forms.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{esāhaṃ bhavantaṃ gotamaṃ saraṇaṃ gacchāmi, dhammañca, bhikkhusaṅghañca.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{I go to Master Gotama for refuge and to the Dhamma and to the Sangha of bhikkhus.}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{upāsakaṃ maṃ bhavaṃ gotamo dhāretu ajjatagge pāṇupetaṃ saraṇaṃ gatan”ti.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{From today let Master Gotama remember me as a lay follower who has gone to him for refuge for life.”}}\\
\end{addmargin}
\end{absolutelynopagebreak}
