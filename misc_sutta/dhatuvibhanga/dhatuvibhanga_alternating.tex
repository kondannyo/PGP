\setstretch{.5}
{\EnglishColumn{and does not yearn.}}\\

\setstretch{.7}
{\PaliColumn{ajīyamāno kiṃ mīyissati,}}\\

\setstretch{.5}

\end{absolutelynopagebreak}

\setstretch{.5}

{\PaliColumn{‘yattha ṭhitaṃ maññassavā nappavattanti, maññassave kho pana nappavattamāne muni santoti vuccatī’ti - iti yaṃ taṃ vuttaṃ, idametaṃ paṭicca vuttaṃ.}}\\

{\EnglishColumn{33. Thereupon the venerable Pukkusāti thought:}}\\

\setstretch{.7}

\begin{absolutelynopagebreak}

\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{“accayo maṃ, bhante, accagamā yathābālaṃ yathāmūḷhaṃ yathāakusalaṃ, yohaṃ bhagavantaṃ āvusovādena samudācaritabbaṃ amaññissaṃ.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}

\begin{absolutelynopagebreak}

\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{“taggha tvaṃ, bhikkhu, accayo accagamā yathābālaṃ yathāmūḷhaṃ yathāakusalaṃ, yaṃ maṃ tvaṃ āvusovādena samudācaritabbaṃ amaññittha.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}
{\EnglishColumn{“Surely, bhikkhu, a transgression overcame you, in that like a fool, confused and blundering, you presumed to address me as ‘friend.’}}\\
\end{addmargin}
\end{absolutelynopagebreak}

\begin{absolutelynopagebreak}
\setstretch{.7}
{\PaliColumn{yato ca kho tvaṃ, bhikkhu, accayaṃ accayato disvā yathādhammaṃ paṭikarosi, taṃ te mayaṃ paṭiggaṇhāma.}}\\
\begin{addmargin}[1em]{2em}
\setstretch{.5}

\begin{addmargin}[1em]{2em}
